26// Represents the memory used by a component. Collects slots that contain
27// components or other frames. This information is later used to generate
28// field reads and writes in the LLZK structs for the operations that are
29// needed in the constrain function but cannot be generated there.
30class Frame {
31public:
32 Frame();
33 Frame(const Frame &);
34 Frame(Frame &&);
35 Frame &operator=(const Frame &);
36 Frame &operator=(Frame &&);
37
38 template <typename Slot, typename... Args> Slot *allocateSlot(Args &&...args) {
39 return static_cast<Slot *>(info->allocateSlot<Slot, Args...>(std::forward<Args>(args)...));
40 }
41
42 using iterator = detail::FrameInfo::iterator;
43 using const_iterator = detail::FrameInfo::const_iterator;
44
45 iterator begin();
46 const_iterator begin() const;
47 iterator end();
48 const_iterator end() const;
49
50 // The frame information forms a tree of slots.
51 // Thus, each frame may be contained inside a slot of a frame in an upper level of the hierarchy.
52 // Once set, overwriting the parent is considered a bug.
53 void setParentSlot(FrameSlot *);
54 FrameSlot *getParentSlot() const;
55
56 friend FrameSlot;
57
58 void print(llvm::raw_ostream &) const;
59
60private:
61 // A pointer to the unique information about the frame
62 std::shared_ptr<detail::FrameInfo> info;
63};
