98void cloneAttrsIntoLlzkFunc(mlir::func::FuncOp src, llzk::function::FuncDefOp dest) {
99 // Add the attributes of this function to dest (except visibility unless is
100 // extern).
101 llvm::MapVector<mlir::StringAttr, mlir::Attribute> newAttrMap;
102 for (const auto &attr : dest->getAttrs()) {
103 newAttrMap.insert({attr.getName(), attr.getValue()});
104 }
105 for (const auto &attr : src->getAttrs()) {
106 newAttrMap.insert({attr.getName(), attr.getValue()});
107 }
108
109 if (!newAttrMap.contains(mlir::StringAttr::get(src.getContext(), "extern"))) {
110 newAttrMap.erase(mlir::StringAttr::get(src.getContext(), "sym_visibility"));
111 }
112
113 auto newAttrs = llvm::to_vector(llvm::map_range(
114 newAttrMap, [](std::pair<mlir::StringAttr, mlir::Attribute> attrPair
115 ) { return mlir::NamedAttribute(attrPair.first, attrPair.second); }
116 ));
117 dest->setAttrs(mlir::DictionaryAttr::get(src.getContext(), newAttrs));
118}

120mlir::LogicalResult FuncOpLowering::matchAndRewrite(
121 mlir::func::FuncOp op, OpAdaptor, mlir::ConversionPatternRewriter &rewriter
122) const {
123
124 auto type = op.getFunctionType();
125 mlir::TypeConverter::SignatureConversion result(type.getNumInputs());
126
127 llvm::SmallVector<mlir::Type, 1> newResults;
128 if (failed(getTypeConverter()->convertSignatureArgs(type.getInputs(), result)) ||
129 failed(getTypeConverter()->convertTypes(type.getResults(), newResults)) ||
130 failed(rewriter.convertRegionTypes(&op.getFunctionBody(), *getTypeConverter(), &result))) {
131 return mlir::failure();
132 }
133
134 auto newType =
135 mlir::FunctionType::get(rewriter.getContext(), result.getConvertedTypes(), newResults);
136
137 auto newFuncOp =
138 rewriter.create<llzk::function::FuncDefOp>(op.getLoc(), op.getNameAttr(), newType);
139 cloneAttrsIntoLlzkFunc(op, newFuncOp);
140 rewriter.inlineRegionBefore(op.getRegion(), newFuncOp.getRegion(), newFuncOp.end());
141 rewriter.replaceOp(op, newFuncOp);
142 return mlir::success();
143}

154) const {

171 op, results, callee.getNameAttr(), adaptor.getCalleeOperands()
172 );
173 return mlir::success();

196// Set of the builtins (by name) that are converted to a LLZK felt or a extended field element
197// representation. Used for shortcircuiting the lowering of SuperCoerceOp and for removing the calls
198// to @constrain that point to structs that get removed during lowering because the implementation
199// of these types gets removed.
200static std::unordered_set<std::string_view> builtinsConvertibleToOps{
201 "Val", "Add", "Sub", "Mul", "BitAnd", "Inv", "Isz", "Neg",
202 "Mod", "InRange", "ExtVal", "ExtAdd", "ExtSub", "ExtInv", "ExtMul", "MakeExt"
203};
204
205static bool wasConvertedToPrimitiveType(ComponentType t) {
206 return t.getBuiltin() &&
207 (builtinsConvertibleToOps.find(t.getName().getValue()) != builtinsConvertibleToOps.end());
208}

210static Value readArray(Value src, Value iv, OpBuilder &builder, Location loc) {
211 auto type = mlir::cast<llzk::array::ArrayType>(src.getType());
212 if (type.getDimensionSizes().size() == 1) {
213 return builder.create<llzk::array::ReadArrayOp>(loc, src, iv);
214 }
215 return builder.create<llzk::array::ExtractArrayOp>(loc, src, iv);
216}
217
218static void writeArray(Value dst, Value iv, Value val, OpBuilder &builder, Location loc) {
219 auto type = mlir::cast<llzk::array::ArrayType>(dst.getType());
220 if (type.getDimensionSizes().size() == 1) {
221 builder.create<llzk::array::WriteArrayOp>(loc, dst, iv, val);
222 } else {
223 builder.create<llzk::array::InsertArrayOp>(loc, dst, iv, val);
224 }
225}

227static Value readSuperFields(
228 ComponentType type, Value chain, Type target, const TypeConverter &tc, Location loc,
229 OpBuilder &builder

232static Value copyArraySuperFields(
233 ComponentType innerType, Value chain, Type target, const TypeConverter &tc, Location loc,
234 OpBuilder &builder
235) {
236 auto targetInner = mlir::cast<ComponentType>(target).getArrayInnerType();
237 auto targetArrayType = mlir::cast<llzk::array::ArrayType>(tc.convertType(target));
238
239 auto array = builder.create<llzk::array::CreateArrayOp>(loc, targetArrayType);
240
241 auto lb = builder.create<arith::ConstantIndexOp>(loc, 0);
242 auto ub = builder.create<llzk::array::ArrayLengthOp>(
243 loc, TypeRange({IndexType::get(builder.getContext())}), chain, lb
244 );
245 auto stride = builder.create<arith::ConstantIndexOp>(loc, 1);
246
247 auto loop = builder.create<scf::ForOp>(
248 loc, lb, stride, ub, ValueRange({array}),
249 [&](OpBuilder &loopBuilder, Location loopLoc, Value iv, ValueRange args) {
250 auto src = readArray(chain, iv, loopBuilder, loopLoc);
251 auto super = readSuperFields(innerType, src, *targetInner, tc, loopLoc, loopBuilder);
252 writeArray(args[0], iv, super, loopBuilder, loopLoc);
253
254 loopBuilder.create<scf::YieldOp>(loopLoc, args);
255 }
256 );
257 return loop.getResult(0);
258}

260static Value handleArraySpecialCases(
261 ComponentType type, Value chain, Type target, const TypeConverter &tc, Location loc,
262 OpBuilder &builder
263) {
264 auto targetAsComp = mlir::dyn_cast_if_present<ComponentType>(target);
265 if (!targetAsComp || !targetAsComp.isConcreteArray()) {
266 return chain;
267 }
268
269 auto targetLlzkType = mlir::cast<llzk::array::ArrayType>(tc.convertType(target));
270 // We don't need to create the copying code if the llzk versions of the types are going to unify.
271 if (llzk::typesUnify(chain.getType(), targetLlzkType)) {
272 // Cast the output type to the equivalent target type to avoid an unrealizable conversion cast.
273 return builder.create<llzk::polymorphic::UnifiableCastOp>(loc, targetLlzkType, chain);
274 }
275
276 // If the target is an array component then read the super fields of the inner elements and
277 // returns a new array with those values.
278 return copyArraySuperFields(
279 mlir::cast<ComponentType>(*type.getArrayInnerType()), chain, target, tc, loc, builder
280 );
281}

283static Value readSuperFields(
284 ComponentType type, Value chain, Type target, const TypeConverter &tc, Location loc,
285 OpBuilder &builder
286) {
287
288 auto superComp = mlir::dyn_cast_if_present<ComponentType>(type.getSuperType());
289 // Stop if we reached a builtin that transforms to a primitive llzk type, we are done, or we
290 // cannot continue extracting
291 if (type == target || wasConvertedToPrimitiveType(type) || !superComp ||
292 mlir::isa<TypeVarType>(type.getSuperType())) {
293 return chain;
294 }
295 if (type.isConcreteArray()) {
296 return handleArraySpecialCases(type, chain, target, tc, loc, builder);
297 }
298
299 auto read = builder.create<llzk::component::FieldReadOp>(
300 loc, tc.convertType(superComp), chain, builder.getStringAttr("$super")
301 );
302 return readSuperFields(superComp, read, target, tc, loc, builder);
303}

320LogicalResult LowerConstrainCallOp::matchAndRewrite(
321 ConstrainCallOp op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter
322) const {
323 auto compType = op.getSelf().getType();
324 auto comp = mlir::dyn_cast<ComponentType>(compType);
325 if (!comp) {
326 return op->emitOpError() << "was expecting a component type but got " << compType;
327 }
328 auto compName = comp.getName().getAttr();
329 if (wasConvertedToPrimitiveType(comp)) {
330 rewriter.eraseOp(op);
331 return success();
332 }
333 auto sym = mlir::SymbolRefAttr::get(
334 compName, {mlir::SymbolRefAttr::get(rewriter.getStringAttr("constrain"))}
335 );
336 rewriter.replaceOpWithNewOp<llzk::function::CallOp>(op, TypeRange(), sym, adaptor.getOperands());
337
338 return success();
339}

352static Value materializeParam(Attribute attr, OpBuilder &builder, Location loc) {
353 if (auto symAttr = mlir::dyn_cast<SymbolRefAttr>(attr)) {
354 auto param = builder.create<llzk::polymorphic::ConstReadOp>(
355 loc, llzk::felt::FeltType::get(builder.getContext()), symAttr.getRootReference()
356 );
357 return builder.create<llzk::cast::FeltToIndexOp>(loc, builder.getIndexType(), param);
358 }
359 if (auto intAttr = mlir::dyn_cast<IntegerAttr>(attr)) {
360 return builder.create<arith::ConstantIndexOp>(loc, llzk::fromAPInt(intAttr.getValue()));
361 }
362 assert(false && "Cannot materialize something that is not a symbol or a literal integer");
363}

379static void collectAffineParams(ArrayRef<Attribute> attrs, SmallVectorImpl<ConstExprAttr> &out) {
380 for (auto attr : attrs) {
381 if (auto param = mlir::dyn_cast<ConstExprAttr>(attr)) {
382 out.push_back(param);
383 }
384 }

387static ValueRange materializeValuesForParam(
388 ConstExprAttr param, ArrayRef<Attribute> sourceParams, SmallVectorImpl<Value> &out,
389 OpBuilder &builder, Location loc
390) {
391 out = llvm::map_to_vector(param.getFormals(), [&](auto formal) {
392 return materializeParam(sourceParams[formal], builder, loc);
393 });
394 return out;
395}

397static void materializeValuesForParams(
398 ArrayRef<ConstExprAttr> affineParams, ArrayRef<Attribute> sourceParams,
399 MutableArrayRef<SmallVector<Value>> mem, OpBuilder &builder, Location loc,
400 SmallVectorImpl<ValueRange> &out
401) {
402 auto res = llvm::map_to_vector(llvm::zip_equal(mem, affineParams), [&](auto in) -> ValueRange {
403 auto [values, param] = in;
404 return materializeValuesForParam(param, sourceParams, values, builder, loc);
405 });
406 out.insert(out.end(), res.begin(), res.end());
407}

409mlir::LogicalResult CallIndirectOpLoweringInCompute::matchAndRewrite(
410 mlir::func::CallIndirectOp op, OpAdaptor adaptor, mlir::ConversionPatternRewriter &rewriter
411) const {
412 auto parent = op->getParentOfType<llzk::function::FuncDefOp>();
413 if (!parent || parent.getName() != "compute") {
414 return failure(); // Don't operate on non compute calls
415 }
416
417 auto callee = mlir::dyn_cast<ConstructorRefOp>(adaptor.getCallee().getDefiningOp());
418 if (!callee) {
419 return failure();
420 }
421
422 Params params{
423 .callee = mlir::cast<ComponentType>(op.getResult(0).getType()).getParams(),
424 .caller = op->getParentOfType<llzk::component::StructDefOp>()
425 .getConstParams()
426 .value_or(rewriter.getArrayAttr({}))
427 .getValue()
428 };
429
430 auto numDeclParams = params.callee.size() - callee.getNumLiftedParams().getZExtValue();
431
432 AffineParams affineParams;
433 collectAffineParams(params.callee.take_front(numDeclParams), affineParams.decl);
434 collectAffineParams(params.callee.drop_front(numDeclParams), affineParams.lifted);
435
436 // Allocate here the values we may generate
437 SmallVector<SmallVector<Value>> mapOperandsMem(affineParams.size());
438 // This idiom does not use any dimensions
439 SmallVector<int32_t> dimsPerMap(affineParams.size(), 0);
440 // And store a ValueRange pointing to the vector here
441 SmallVector<ValueRange> mapOperands;
442 mapOperands.reserve(affineParams.size());
443
444 materializeValuesForParams(
445 affineParams.decl, params.caller,
446 MutableArrayRef(mapOperandsMem).take_front(affineParams.decl.size()), rewriter, op.getLoc(),
447 mapOperands
448 );
449 materializeValuesForParams(
450 affineParams.lifted, params.callee,
451 MutableArrayRef(mapOperandsMem).drop_front(affineParams.decl.size()), rewriter, op.getLoc(),
452 mapOperands
453 );
454
455 auto sym = mlir::SymbolRefAttr::get(
456 callee.getComponentAttr().getAttr(), {mlir::SymbolRefAttr::get(parent.getNameAttr())}
457 );
458
459 llvm::SmallVector<mlir::Type> types;
460 auto convRes = getTypeConverter()->convertTypes(op.getResultTypes(), types);
461 if (mlir::failed(convRes)) {
462 return op->emitError("failed to transform zml types into llzk types");
463 }
464 mlir::ValueRange args(
465 mlir::iterator_range(adaptor.getOperands().begin() + 1, adaptor.getOperands().end())
466 );
467
468 rewriter.replaceOpWithNewOp<llzk::function::CallOp>(
469 op, types, sym, mapOperands, dimsPerMap, args
470 );
471 return mlir::success();
472}

516mlir::LogicalResult LowerInRangeOp::matchAndRewrite(
517 InRangeOp op, OpAdaptor adaptor, mlir::ConversionPatternRewriter &rewriter
518) const {
519 auto le = rewriter.create<llzk::boolean::CmpOp>(
520 op.getLoc(),
521 llzk::boolean::FeltCmpPredicateAttr::get(getContext(), llzk::boolean::FeltCmpPredicate::LE),
522 adaptor.getLow(), adaptor.getMid()
523 );
524 auto lt = rewriter.create<llzk::boolean::CmpOp>(
525 op.getLoc(),
526 llzk::boolean::FeltCmpPredicateAttr::get(getContext(), llzk::boolean::FeltCmpPredicate::LT),
527 adaptor.getMid(), adaptor.getHigh()
528 );
529 auto felt = llzk::felt::FeltType::get(getContext());
530 auto convLe = rewriter.create<llzk::cast::IntToFeltOp>(op.getLoc(), felt, le);
531 auto convLt = rewriter.create<llzk::cast::IntToFeltOp>(op.getLoc(), felt, lt);
532 auto mul = rewriter.create<llzk::felt::MulFeltOp>(op.getLoc(), convLe, convLt);
533 rewriter.replaceOp(op, mul);
534
535 return mlir::success();
536}

538mlir::LogicalResult LowerNewArrayOp::matchAndRewrite(
539 NewArrayOp op, OpAdaptor adaptor, mlir::ConversionPatternRewriter &rewriter
540) const {
541 // NewArrayOp is SameTypeOperands so by querying the type of any one of the elements we can know
542 // if the inputs are scalar or arrays
543
544 auto type = getTypeConverter()->convertType(op.getType());
545 auto arrType = dyn_cast<llzk::array::ArrayType>(type);
546 if (!arrType) {
547 return op.emitOpError() << "was expecting an array type";
548 }
549
550 // If it's an array then we allocate an empty one and then insert each operand with InsertArrayOp
551 if (!adaptor.getElements().empty() &&
552 isa<llzk::array::ArrayType>(adaptor.getElements()[0].getType())) {
553 Location cachedLoc = op.getLoc();
554 auto arr = rewriter.replaceOpWithNewOp<llzk::array::CreateArrayOp>(op, arrType, ValueRange());
555 for (size_t i = 0; i < adaptor.getElements().size(); i++) {
556 auto idx = rewriter.create<arith::ConstantIndexOp>(cachedLoc, i);
557 rewriter.create<llzk::array::InsertArrayOp>(
558 cachedLoc, arr, ValueRange({idx}), adaptor.getElements()[i]
559 );
560 }
561 } else {
562 rewriter.replaceOpWithNewOp<llzk::array::CreateArrayOp>(op, arrType, adaptor.getElements());
563 }
564 return mlir::success();
565}

587mlir::LogicalResult LowerReadArrayOp::matchAndRewrite(
588 ReadArrayOp op, OpAdaptor adaptor, mlir::ConversionPatternRewriter &rewriter
589) const {
590 llvm::SmallVector<mlir::Value> toIndexOps =
591 llvm::map_to_vector(adaptor.getIndices(), [&](mlir::Value index) -> mlir::Value {
592 if (mlir::isa<mlir::IndexType>(index.getType())) {
593 return mlir::cast<mlir::TypedValue<mlir::IndexType>>(index);
594 }

627mlir::LogicalResult LowerAllocArrayOp::matchAndRewrite(
628 AllocArrayOp op, OpAdaptor, mlir::ConversionPatternRewriter &rewriter
629) const {
630 auto type = getTypeConverter()->convertType(op.getType());
631 auto arrType = dyn_cast<llzk::array::ArrayType>(type);
632 if (!arrType) {
633 return op.emitOpError() << "was expecting an array type";
634 }
635
636 ArrayAttr compParams = op->getParentOfType<llzk::component::StructDefOp>().getType().getParams();
637 SmallVector<ConstExprAttr> affineArrayParams;
638 ComponentType cType = op.getResult().getType();
639 for (Attribute attr : cType.getParams()) {
640 if (auto param = mlir::dyn_cast<ConstExprAttr>(attr)) {
641 affineArrayParams.push_back(param);
642 }
643 }
644 // Allocate here the values we may generate
645 SmallVector<SmallVector<Value>> mapOperandsMem(affineArrayParams.size());
646 // And store a ValueRange pointing to the vector here
647 SmallVector<ValueRange> mapOperands;
648 // This idiom does not use any dimensions
649 SmallVector<int32_t> dimsPerMap(affineArrayParams.size(), 0);
650 mapOperands.reserve(affineArrayParams.size());
651
652 for (auto [idx, constExpr] : llvm::enumerate(affineArrayParams)) {
653 auto &values = mapOperandsMem[idx];
654 for (uint64_t formal : constExpr.getFormals()) {
655 assert(formal <= std::numeric_limits<unsigned int>::max());
656 if (static_cast<unsigned int>(formal) >= compParams.size()) {
657 return op->emitError() << "requested parameter #" << formal
658 << ", but component in scope only has " << compParams.size()
659 << " parameters";
660 }
661 values.push_back(
662 materializeParam(compParams[static_cast<unsigned int>(formal)], rewriter, op->getLoc())
663 );
664 }
665 mapOperands.push_back(values);
666 }
667
668 rewriter.replaceOpWithNewOp<llzk::array::CreateArrayOp>(op, arrType, mapOperands, dimsPerMap);
669
670 return mlir::success();
671}

780LogicalResult LowerReadBackOp::matchAndRewrite(
781 ReadBackOp op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter
782) const {
783 const auto *converter = getTypeConverter();
784 assert(converter);
785
786 auto replace = [&](auto... args) {
787 auto outType = converter->convertType(op.getResult().getType());
788 rewriter.replaceOpWithNewOp<llzk::component::FieldReadOp>(
789 op, outType, adaptor.getComp(), adaptor.getFieldAttr(),
790 std::forward<decltype(args)>(args)...
791 );
792 };
793 llvm::TypeSwitch<Attribute>(adaptor.getDistance())
794 .Case([&](SymbolRefAttr symAttr) {
795 // If the distance is a symbol create an affine expression that negates the symbol.
796 // Expects the symbol to be one of the parameters of the current struct. Will generate malformed
797 // IR otherwise.
798 replace(
799 AffineMapAttr::get(AffineMap::get(
800 /*dimCount=*/0, /*symbolCount=*/1,
801 rewriter.getAffineConstantExpr(0) - rewriter.getAffineSymbolExpr(0)
802 )),
803 // materializeParam's Value return will be implicitly cast to a ValueRange.
804 // The previous creation of a local variable `ValueRange mapOperands({value})`
805 // causes errors as ValueRange is a thin wrapper.
806 materializeParam(symAttr, rewriter, op->getLoc()), 0
807 );
808 })
809 .Case([&](IntegerAttr intAttr) {
810 // If the distance is a literal integer just negate it to move backwards.
811 auto intValue = intAttr.getValue();
812 assert(intValue.isNonNegative());
813 if (!intValue.isZero()) {
814 intValue.negate();
815 }
816 replace(IntegerAttr::get(rewriter.getIndexType(), intValue));
817 })
818 .Case([&](ConstExprAttr cexpAttr) {
819 // If the distance is an expression, wrap it around a negation.
820 auto templMap = AffineMap::get(
821 /*dimCount=*/1, /*symbolCount=*/0,
822 rewriter.getAffineConstantExpr(0) - rewriter.getAffineDimExpr(0)
823 );
824 auto newMap = templMap.compose(cexpAttr.getMap());
825 auto newCexpr = ConstExprAttr::get(newMap, cexpAttr.getFormals());
826 auto params = op->getParentOfType<llzk::component::StructDefOp>()
827 .getConstParams()
828 .value_or(rewriter.getArrayAttr({}))
829 .getValue();
830 SmallVector<SmallVector<Value>, 1> mapOperandsMem(1);
831 SmallVector<ValueRange, 1> mapOperands;
832 materializeValuesForParams(
833 {newCexpr}, params, MutableArrayRef(mapOperandsMem), rewriter, op.getLoc(), mapOperands
834 );
835 replace(Attribute(AffineMapAttr::get(newCexpr.getMap())), mapOperands[0], 0);
836 }).Default([&](Attribute) { llvm_unreachable("ReadBackOp with an unexpected attribute type"); });
837 return success();
838}
