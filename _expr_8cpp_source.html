114mlir::FailureOr<ConstExpr> ConstExpr::remap(Params params, EmitErrorFn emitError) const {
115 // If this pointer is "null" return a null pointer as well.
116 if (!expr) {
117 return ConstExpr(nullptr);
118 }
119
120 auto *replacement = expr->remap(params, emitError);
121 if (!replacement) {
122 return failure();
123 }
124 return ConstExpr(replacement);

164ExprBase *Symbol::remap(Params params, EmitErrorFn emitError) const {
165 auto *binding = params[name];
166 if (!binding) {
167 return clone();
168 }
169 if (!binding->hasConstExpr()) {
170 emitError() << "was expecting a constant expression but got '" << *binding << "'";
171 return nullptr;
172 }
173 return binding->getConstExpr()->clone();
174}

210static bool isInfixExpr(const ExprBase &expr) {
211 if (auto ctor = mlir::dyn_cast<Ctor>(&expr)) {

217static void printSide(llvm::raw_ostream &os, const ExprBase &expr) {
218 bool needsParen = isInfixExpr(expr);

228void Ctor::print(llvm::raw_ostream &os) const {
229 if (isInfixExpr(*this)) {
230 printSide(os, args[0]);
231 os << " " << getInfixSymbol(typeName) << " ";
232 printSide(os, args[1]);
233 } else if (getTypeName() == "Neg") {
234 os << "-";
235 printSide(os, args[0]);
236 } else {
237 os << typeName << "(";
238 llvm::interleaveComma(args, os, [&](auto &e) { e.print(os); });
239 os << ")";
240 }
241}

249Attribute Ctor::convertIntoAttribute(Builder &builder) const {
250 auto expr = convertIntoAffineExpr(builder);
251 if (failed(expr)) {
252 return nullptr;
253 }
254 LLVM_DEBUG(llvm::dbgs() << "Generated expression: " << expr << "\n");
255
256 std::optional<unsigned int> largest;
257 expr->walk([&](AffineExpr e) {
258 if (!mlir::isa<AffineSymbolExpr>(e)) {
259 return;
260 }
261 auto symExpr = mlir::cast<AffineSymbolExpr>(e);
262 largest = std::max(largest.value_or(0), symExpr.getPosition());
263 });
264 unsigned int symbolCount = 0;
265 if (largest.has_value()) {
266 symbolCount = *largest + 1;
267 }
268 auto map = AffineMap::get(0, symbolCount, *expr);
269
270 // After constructing the map reduce the number of symbols and compute what formals they refer to
271 SmallVector<uint64_t> formals;
272 SmallVector<AffineExpr> formalsToSymbols;
273 unsigned int shift = 0;
274 for (unsigned int i = 0; i < map.getNumSymbols(); i++) {
275 if (map.isFunctionOfSymbol(i)) {
276 formals.push_back(i);
277 // Map the i-th formal to the position the symbol will have in the affine map after
278 // removing dead symbols
279 formalsToSymbols.push_back(builder.getAffineSymbolExpr(i - shift));
280 } else {
281 shift++;
282 formalsToSymbols.push_back(builder.getAffineConstantExpr(0));
283 }
284 }
285 assert(formalsToSymbols.size() == map.getNumSymbols());
286 assert(formals.size() <= std::numeric_limits<unsigned>::max());
287 map = map.replaceDimsAndSymbols({}, formalsToSymbols, 0, static_cast<unsigned>(formals.size()));
288 return zml::ConstExprAttr::get(map, formals);
289}

291ExprBase *Ctor::remap(Params params, EmitErrorFn emitError) const {
292 auto replacedArgs =
293 llvm::map_to_vector(args, [&](const ExprBase &arg) { return arg.remap(params, emitError); });
294 if (llvm::any_of(replacedArgs, [](auto &arg) { return !arg; })) {
295 return nullptr;
296 }
297 return new Ctor(typeName, Arguments(std::move(replacedArgs)));

338bool Ctor::Arguments::operator==(const Arguments &other) const {
339 if (lst.size() != other.lst.size()) {
340 return false;
341 }
342
343 for (auto [lhs, rhs] : llvm::zip_equal(lst, other.lst)) {
344 if (lhs != rhs) {
345 return false;
346 }
347 }
348 return true;
