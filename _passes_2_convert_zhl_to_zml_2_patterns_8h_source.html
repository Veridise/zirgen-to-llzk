98 mlir::FailureOr<mlir::Value> getCastedValue(
99 mlir::Value value, mlir::OpBuilder &builder,
100 mlir::SmallVector<mlir::Operation *, 2> &generatedOps, mlir::Type super = nullptr
101 ) const {

109 mlir::FailureOr<mlir::Value>
110 getCastedValue(mlir::Value value, mlir::OpBuilder &builder, mlir::Type super = nullptr) const {
111 auto binding = getType(value);
112 if (mlir::failed(binding)) {
113 return mlir::failure();
114 }
115 mlir::SmallVector<mlir::Operation *, 2> generatedOps;
116 return getCastedValue(value, *binding, builder, generatedOps, super);

118
120 mlir::Value getCastedValue(
121 mlir::Value value, const zhl::TypeBinding &binding, mlir::OpBuilder &builder,
122 mlir::SmallVector<mlir::Operation *, 2> &generatedOps, mlir::Type super = nullptr
123 ) const {
124 auto materialized = materializeTypeBinding(builder.getContext(), binding);
125 assert(materialized);
126 if (value.getType() == materialized) {
127 return value;
128 }
129 auto cast = builder.create<mlir::UnrealizedConversionCastOp>(
130 value.getLoc(), mlir::TypeRange(materialized), mlir::ValueRange(value)
131 );
132 generatedOps.push_back(cast.getOperation());
133
134 mlir::Value result = cast.getResult(0);
135 if (super && super != materialized) {
136 auto coerce = builder.create<SuperCoerceOp>(value.getLoc(), super, result);
137 generatedOps.push_back(coerce.getOperation());
138 result = coerce;
139 }
140 return result;
141 }
142
143 mlir::Value getCastedValue(
144 mlir::Value value, const zhl::TypeBinding &binding, mlir::OpBuilder &builder,
145 mlir::Type super = nullptr
146 ) const {
147 mlir::SmallVector<mlir::Operation *, 2> generatedOps;
148 return getCastedValue(value, binding, builder, generatedOps, super);
149 }

153protected:
