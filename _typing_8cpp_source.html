51class BlockScopesGuard {

96 void checkComponent(ComponentOp comp) {
97 LLVM_DEBUG(llvm::dbgs() << "\n"; loggingScope = comp.getName();
98 logLine() << "Checking component " << comp.getName() << ":\n"; ident++);
99 ComponentScope scope(comp, getTypeBindings());
100 for (auto &op : comp.getRegion().getOps()) {
101 (void)typeCheckOp(&op, scope);
102 }
103 LLVM_DEBUG(ident--; logLine() << "Finished checking component " << comp.getName() << "\n";
104 loggingScope = "");
105 }
106
107private:
108 void markColumn(TypeBinding &binding) {
109 assert(binding.getSlot());
110 mlir::cast<ComponentSlot>(binding.getSlot())->markColumn();
111 }
112
116 void
117 markPotentialColumn(TypeBinding &binding, llvm::function_ref<InFlightDiagnostic()> emitError) {
118 if (!binding.getSlot()) {
119 return;
120 }
121 auto NondetReg = getTypeBindings().MaybeGet("NondetReg");
122 if (failed(NondetReg)) {
123 return;
124 }
125 auto NondetExtReg = getTypeBindings().MaybeGet("NondetExtReg");
126 if (failed(NondetExtReg)) {
127 return;
128 }
129
130 auto validSubtype = [&NondetReg, &NondetExtReg](const TypeBinding &Binding) {
131 return succeeded(Binding.subtypeOf(*NondetReg)) ||
132 succeeded(Binding.subtypeOf(*NondetExtReg));
133 };
134
135 if (validSubtype(binding)) {
136 markColumn(binding);
137 }
138
139 if (binding.isArray()) {
140 auto elt = binding.getArrayElement(emitError);
141 assert(succeeded(elt));
142 if (validSubtype(*elt)) {
143 markColumn(binding);
144 }
145 }
146
147 const auto *b = &binding;
148 while (b) {
149 b = [&]() -> const TypeBinding * {
150 for (auto &member : b->getMembers()) {
151 if (member.getValue().has_value() && validSubtype(*member.getValue())) {
152 markColumn(binding);
153 return nullptr;
154 }
155 }
156 if (b->hasSuperType()) {
157 return &b->getSuperType();
158 }
159 return nullptr;
160 }();
161 }
162 }

164 FailureOr<TypeBinding> typeCheckOp(Operation *op, Scope &scope) {
165 assert(op != nullptr);
166 LLVM_DEBUG(logLine() << "Checking " << op->getName() << "\n";
167 if (op->getNumRegions() == 0) { logLine() << " " << *op << "\n"; });
168 if (getBindings().contains(op)) {
169 auto &cachedBinding = getBindings().get(op);
170 LLVM_DEBUG(logLine() << "Pulled from cache: "; if (failed(cachedBinding)) {
171 llvm::dbgs() << "<<FAILURE>>\n";
172 } else { llvm::dbgs() << *cachedBinding << "\n"; });
173 return cachedBinding;
174 }
175 LLVM_DEBUG(ident++);
176 auto operands = getOperandTypes(op, scope);
177 LLVM_DEBUG(ident--);
178 if (failed(operands)) {
179 LLVM_DEBUG(logLine() << "Failed to obtain bindings of the operation's operands\n");
180 return failure();
181 }
182 for (auto &rule : getRules()) {
183 if (failed(rule->match(op))) {
184 continue;
185 }
186 LLVM_DEBUG(llvm::dbgs() << "\n");
187 auto frame = rule->allocate(scope.getCurrentFrame());
188 FailureOr<TypeBinding> ruleResult =
189 succeeded(frame) ? applyRuleWithFrame(*frame, op, scope, *rule, *operands)
190 : applyRule(op, scope, *rule, *operands);
191
192 LLVM_DEBUG(llvm::dbgs() << "\n");
193 if (succeeded(ruleResult)) {
194 LLVM_DEBUG(logLine() << "SUCCESS: Storing binding "; ruleResult->print(llvm::dbgs(), true);
195 llvm::dbgs() << "\n");
196 markPotentialColumn(*ruleResult, [op] { return op->emitError(); });
197 return getBindings().add(op, ruleResult);
198 }
199 }
200 LLVM_DEBUG(logLine() << " FAILURE: Not rule was matched\n");
201 return getBindings().add(
202 op, op->emitError() << "could not deduce the type of op '" << op->getName() << "'"
203 );
204 }
205
206 FailureOr<std::vector<TypeBinding>> getOperandTypes(Operation *op, Scope &scope) {
207 auto operands = op->getOperands();
208 std::vector<TypeBinding> operandBindings;
209 for (auto operand : operands) {
210 auto r = typeCheckValue(operand, scope);
211 if (failed(r)) {
212 return failure();
213 }
214 operandBindings.push_back(*r);
215 }
216 return operandBindings;
217 }
218
219 FailureOr<TypeBinding> typeCheckValue(Value v, Scope &scope) {
220 if (Operation *op = v.getDefiningOp()) {
221 return typeCheckOp(op, scope);
222 }
223 if (getBindings().containsValue(v)) {
224 return getBindings().getValue(v);
225 }
226 return failure();
227 }

229 FailureOr<TypeBinding>
230 applyRule(Operation *op, Scope &scope, const TypingRule &rule, ArrayRef<TypeBinding> operands) {
231 // Check if the operation has regions and evaluate them first
232 auto regions = op->getRegions();
233 BlockScopesGuard regionScopes(scope, getTypeBindings());
234
235 if (!std::all_of(regions.begin(), regions.end(), [&](auto &region) {
236 auto valueBindings =
237 rule.bindRegionArguments(ValueRange(region.getArguments()), op, operands, scope);
238 if (failed(valueBindings)) {
239 return false;
240 }
241 for (auto [value, type] : llvm::zip_equal(
242 mlir::iterator_range(region.args_begin(), region.args_end()), *valueBindings
243 )) {
244 (void)getBindings().addValue(value, type);
245 }
246 return succeeded(typeCheckRegion(region, regionScopes.get()));
247 })) {
248 return failure();
249 }
250
251 return rule.typeCheck(op, operands, scope, regionScopes);
252 }
255 FailureOr<TypeBinding> applyRuleWithFrame(
256 Frame frame, Operation *op, Scope &scope, const TypingRule &rule,
257 ArrayRef<TypeBinding> operands
258 ) {
259 FrameScope frameScope(scope, frame);
260 // Typecheck normally using a scope with the frame
261 auto result = applyRule(op, frameScope, rule, operands);
262
263 if (failed(result)) {
264 return failure();
265 }
266 auto finalBinding = TypeBinding::ReplaceFrame(*result, frame);
267 auto *parent = frame.getParentSlot();
268 if (auto *compParent = mlir::dyn_cast_if_present<ComponentSlot>(parent)) {
269 if (mlir::isa_and_present<ComponentSlot>(finalBinding.getSlot())) {
270 // Remove the slot if it is a ComponentSlot since we are
271 // going to store that component in the parent slot instead
272 finalBinding.markSlot(nullptr);
273 }
274 compParent->setBinding(finalBinding);
275 }
276 return finalBinding;
277 }
278
279 LogicalResult typeCheckRegion(Region &region, Scope &scope) {
280 bool anyFailed = false;
281
282 for (auto &op : region.getOps()) {
283 auto result = typeCheckOp(&op, scope);
284 anyFailed = anyFailed || failed(result);
285 }
286
287 return anyFailed ? failure() : success();
288 }

294class ZhlSymbolTable {
295public:
296 explicit ZhlSymbolTable(Operation *op) : st(op) {}

298 ComponentOp lookup(StringRef name) const {
299 for (auto &region : st->getRegions()) {
300 auto I = region.op_begin<ComponentOp>();
301 auto E = region.op_end<ComponentOp>();
302 for (; I != E; ++I) {
303 auto op = *I;
304 if (op.getName() == name) {
305 return op;
306 }
307 }
308 }
309 return nullptr;
310 }

318void fillGraph(Operation *root, GraphInDegrees &indegrees) {
319 root->walk([&](ComponentOp op) { indegrees[op.getName()]; });
320}

327LogicalResult
328sortComponents(Operation *root, SmallVector<ComponentOp> &sortedComponents, ZhlSymbolTable &st) {
329 assert(sortedComponents.empty() && "output vector must be empty");
330 GraphInDegrees indegrees;
331 fillGraph(root, indegrees);
332 root->walk([&](ComponentOp op) {
333 op->walk([&](GlobalOp global) {
334 // We only add to the graph the component if its a name declared as a ZHL component op.
335 // This excludes builtins that have bindings but not a ZHL representation because they are
336 // generated directly in ZML.
337 if (indegrees.contains(global.getName())) {
338 indegrees[global.getName()]++;
339 }
340 });
341 });
342 // To avoid unnecessary allocations
343 sortedComponents.reserve(indegrees.size());
344
345 // Prepare the initial queue for Kahn's algorithm with the components with indegree of 0
346 SmallVector<ComponentOp> queue;
347 for (auto &[name, indegreeCount] : indegrees) {
348 if (indegreeCount == 0) {
349 auto compOp = st.lookup(name);
350 assert(compOp);
351 queue.push_back(compOp);
352 }
353 }
354
355 // Kahn's algorithm will find the topological order or fail if there is a cycle.
356 while (!queue.empty()) {
357 auto op = queue.front();
358 queue.erase(queue.begin());
359 // Since its traverses the graph in reverse we need to add the elements to front of the vector
360 sortedComponents.insert(sortedComponents.begin(), op);
361
362 op->walk([&](GlobalOp global) {
363 if (indegrees.contains(global.getName())) {
364 indegrees[global.getName()]--;
365 assert(indegrees[global.getName()] >= 0 && "More edges were removed than inserted");
366 if (indegrees[global.getName()] == 0) {
367 auto depOp = st.lookup(global.getName());
368 assert(depOp);
369 queue.push_back(depOp);
370 }
371 }
372 });
373 }
374
375 // Check that all indegrees were remove (aka no cycles)
376 for (auto &[name, indegree] : indegrees) {
377 if (indegree > 0) {
378 return root->emitError() << "Recursion cycle detected in component " << name;
379 }
380 }
381
382 return success();
383}

387FailureOr<std::unique_ptr<ZhlOpBindings>>
388typeCheck(Operation *root, TypeBindings &typeBindings, const FrozenTypingRuleSet &rules) {
389 // Initialize what we need for type checking
390 SymbolTable st(root);
391 ZhlSymbolTable zhlSt(root);
392 auto bindings = std::make_unique<ZhlOpBindings>();
393 SmallVector<ComponentOp> sortedComponents;
394 if (failed(sortComponents(root, sortedComponents, zhlSt))) {
395 return failure();
396 }
397 // Run the typechecker
398 TypeChecker tc(*bindings, rules, typeBindings);
399 for (auto op : sortedComponents) {
400 tc.checkComponent(op);
401 }
402
403 return bindings;
404}
