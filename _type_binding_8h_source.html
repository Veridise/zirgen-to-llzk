124 class Flags {
125 enum : unsigned {
128 Variadic,
131 Specialized,
133 SelfConstructor,
135 Builtin,
138 Closure,
140 Extern,
142 NeedsBackVariables,
143
144 Flags_End
145 };
146
147 using bitset = llvm::Bitset<Flags_End>;
148
149 bitset flags;
150
151 constexpr void set(unsigned idx, bool value) {
152 if (value) {
153 flags.set(idx);
154 } else {
155 flags.reset(idx);
156 }
157 }
158
159 public:
160 constexpr Flags() {}
161 constexpr Flags(std::initializer_list<unsigned> Init) : flags(Init) {}
162
163 static constexpr Flags MkBuiltin(bool builtin) {
164 Flags f;
165 f.setBuiltin(builtin);
166 return f;
167 }
168
169 constexpr bool operator==(const Flags &other) const { return flags == other.flags; }
170
172 constexpr bool isVariadic() const { return flags[Variadic]; }

175 set(Variadic, B);
176 return *this;
