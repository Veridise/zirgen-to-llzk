36mlir::Operation *findCallee(mlir::StringRef name, mlir::ModuleOp root) {
37 auto calleeComp = root.lookupSymbol<ComponentInterface>(name);
38 if (calleeComp) {
39 return calleeComp;
40 }
41
42 // Zhl Component ops don't declare their symbols in the symbol table
43 for (auto zhlOp : root.getOps<zirgen::Zhl::ComponentOp>()) {
44 if (zhlOp.getName() == name) {
45 return zhlOp;
46 }
47 }
48 return nullptr;
49}

64FlatSymbolRefAttr createSlot(
65 zhl::ComponentSlot *slot, OpBuilder &builder, ComponentInterface component, Location loc
66) {
67 mlir::SymbolTable st(component);
68
69 auto desiredName = mlir::StringAttr::get(component.getContext(), slot->getSlotName());
70 LLVM_DEBUG(
71 llvm::dbgs() << "Creating a slot in " << component.getName()
72 << "\nDesired name: " << desiredName << "\n"
73 );
74 mlir::OpBuilder::InsertionGuard guard(builder);
75 builder.setInsertionPointAfter(&component.getRegion().front().front());
76 auto type = materializeTypeBinding(builder.getContext(), slot->getBinding());
77 auto fieldDef = builder.create<FieldDefOp>(
78 loc, desiredName, TypeAttr::get(type), slot->isColumn() ? builder.getUnitAttr() : nullptr
79 );
80
81 LLVM_DEBUG(llvm::dbgs() << "Field op: " << fieldDef << "\n");
82 // Insert the FieldDefOp into the symbol table to make sure it has an unique name within the
83 // component
84 auto name = st.insert(fieldDef);
85 LLVM_DEBUG(llvm::dbgs() << "Name created by the symbol table: " << name << "\n");
86 slot->rename(name.getValue());
87 return mlir::FlatSymbolRefAttr::get(component.getContext(), name);
88}

90TypeBinding unwrapArrayNTimes(const TypeBinding &type, size_t count) {
91 if (count == 0) {
92 return type;
93 }
94 assert(type.isArray());
95 auto inner = type.getArrayElement([]() { return mlir::InFlightDiagnostic(); });
96 assert(succeeded(inner));
97 return unwrapArrayNTimes(*inner, count - 1);
98}

108Value loadSlot(
109 ComponentSlot &slot, Type valueType, FlatSymbolRefAttr slotName, Type slotType, Location loc,
110 OpBuilder &builder, Value self
111) {
112 auto compSlotBinding = slot.getBinding();
113 auto compSlotIVs = slot.collectIVs();
114
115 if (compSlotIVs.empty()) {
116 // Read the temporary back to a SSA value
117 return builder.create<ReadFieldOp>(loc, slotType, self, slotName);
118 } else {
119 // Read the array back to a SSA value
120 auto arrayDataBis = builder.create<ReadFieldOp>(loc, slotType, self, slotName);
121
122 // Read the value we wrote into the array back to a SSA value
123 return builder.create<ReadArrayOp>(loc, valueType, arrayDataBis, compSlotIVs);
124 }
125}

127void storeSlot(
128 ComponentSlot &slot, Value value, FlatSymbolRefAttr slotName, Type slotType, Location loc,
129 OpBuilder &builder, Value self
130) {
131 auto compSlotBinding = slot.getBinding();
132 auto compSlotIVs = slot.collectIVs();
133
134 if (compSlotIVs.empty()) {
135 LLVM_DEBUG(
136 llvm::dbgs() << "slotType == " << slotType << " | value.getType() == " << value.getType()
137 << "\n"
138 );
139 assert(slotType == value.getType() && "result of construction and slot type must be the same");
140 // Write the construction in a temporary
141 builder.create<WriteFieldOp>(loc, self, slotName, value);
142 } else {
143 auto unwrappedBinding = unwrapArrayNTimes(compSlotBinding, compSlotIVs.size());
144 auto unwrappedType = materializeTypeBinding(builder.getContext(), unwrappedBinding);
145 assert(
146 unwrappedType == value.getType() &&
147 "result of construction and slot inner array type must be the same"
148 );
149 // Read the array from the slot field
150 auto arrayData = builder.create<ReadFieldOp>(loc, slotType, self, slotName);
151 assert(arrayData);
152 assert(value);
153 // Write into the array the value
154 builder.create<WriteArrayOp>(loc, arrayData, compSlotIVs, value, true);
155
156 // Write the array back into the field
157 builder.create<WriteFieldOp>(loc, self, slotName, arrayData);
158 }
159}

163void materializeFieldTypes(
164 zhl::TypeBinding &binding, MLIRContext *ctx, SmallVectorImpl<Type> &types,
165 SmallVectorImpl<StringRef> &fields, SmallVectorImpl<bool> *columns = nullptr
166) {
167 llvm::StringMap<std::pair<Type, bool>> map;
168 SmallVector<StringRef> fieldsToSort;
169 fieldsToSort.reserve(binding.getMembers().size());
170
171 for (auto &[memberName, memberBinding] : binding.getMembers()) {
172 assert(memberBinding.has_value());
173 auto memberType = materializeTypeBinding(ctx, *memberBinding);
174 bool isColumn = false;
175 if (auto *cslot = mlir::dyn_cast_if_present<ComponentSlot>(memberBinding->getSlot())) {
176 isColumn = cslot->isColumn();
177 }
178 map[memberName] = {memberType, isColumn};
179 fieldsToSort.push_back(memberName);
180 }
181
182 std::sort(fieldsToSort.begin(), fieldsToSort.end());
183 for (auto field : fieldsToSort) {
184 types.push_back(map[field].first);
185 if (columns) {
186 columns->push_back(map[field].second);
187 }
188 }
189 fields.insert(fields.end(), fieldsToSort.begin(), fieldsToSort.end());
190}

192void constructFieldReads(
193 TypeRange types, ArrayRef<StringRef> fieldNames, SmallVectorImpl<Value> &results,
194 OpBuilder &builder, Value self, Location loc, zhl::TypeBinding &binding
195) {
196 for (auto [type, field] : llvm::zip_equal(types, fieldNames)) {
197 auto memberBinding = binding.getMembers().at(field);
198 assert(memberBinding.has_value());
199 ComponentSlot *slot = mlir::dyn_cast_if_present<ComponentSlot>(memberBinding->getSlot());
200 assert(slot);
201 auto slotName = FlatSymbolRefAttr::get(builder.getStringAttr(slot->getSlotName()));
202 auto slotType = materializeTypeBinding(builder.getContext(), slot->getBinding());
203 results.push_back(loadSlot(*slot, type, slotName, slotType, loc, builder, self));
204 }
205}

207FailureOr<Value> constructPODComponent(
208 Operation *op, zhl::TypeBinding &binding, OpBuilder &builder, Value self,
209 llvm::function_ref<mlir::Value()> superTypeValueCb, const zhl::TypeBindings &bindings
210) {
211 auto ctor = CtorCallBuilder::Make(op, binding, builder, self, bindings);
212 if (failed(ctor)) {
213 return failure();
214 }
215
216 auto loc = binding.getLocation();
217
218 auto superTypeValue = superTypeValueCb();
219 SmallVector<Type, 1> argTypes({superTypeValue.getType()});
220 SmallVector<StringRef, 1> fieldNames({"$super"});
221 SmallVector<Value, 1> args({superTypeValue});
222 auto toReserve = 1 + binding.getMembers().size();
223 argTypes.reserve(toReserve);
224 fieldNames.reserve(toReserve);
225 args.reserve(toReserve);
226
227 materializeFieldTypes(binding, builder.getContext(), argTypes, fieldNames);
228 constructFieldReads(
229 ArrayRef(argTypes).drop_front(), ArrayRef(fieldNames).drop_front(), args, builder, self, loc,
230 binding
231 );
232
233 return ctor->build(builder, loc, args);
234}

236void createPODComponent(
237 zhl::TypeBinding &binding, mlir::OpBuilder &builder, mlir::SymbolTable &st
238) {
239 SmallVector<Type> argTypes;
240 SmallVector<StringRef> fieldNames;
241 SmallVector<bool> columns;
242
243 ComponentBuilder cb;
244 auto loc = binding.getLocation();
245 cb.name(binding.getName()).isClosure().location(loc);
246
247 auto superType = materializeTypeBinding(builder.getContext(), binding.getSuperType());
248 argTypes = {superType};
249 fieldNames = {"$super"};
250 columns = {false};
251 argTypes.reserve(1 + binding.getMembers().size());
252 fieldNames.reserve(1 + binding.getMembers().size());
253 columns.reserve(1 + binding.getMembers().size());
254
255 materializeFieldTypes(binding, builder.getContext(), argTypes, fieldNames, &columns);
256 for (auto [memberName, memberType, isColumn] : llvm::zip_equal(fieldNames, argTypes, columns)) {
257 cb.field(memberName, memberType, isColumn);
258 }
259
260 if (!binding.getGenericParamNames().empty()) {
261 cb.forceGeneric().typeParams(binding.getGenericParamNames());
262 }
263 cb.defer([&](ComponentOp compOp) {
264 auto actualName = st.insert(compOp);
265 binding.setName(actualName.getValue());
266
267 // Now that we know the final name of the component we can configure the rest
268 cb.fillBody(
269 argTypes, {materializeTypeBinding(builder.getContext(), binding)},
270 [&](mlir::ValueRange args, mlir::OpBuilder &B) {
271 auto componentType = materializeTypeBinding(B.getContext(), binding);
272 // Reference to self
273 auto self = B.create<SelfOp>(loc, componentType);
274 // Store the fields
275 for (auto [fieldName, arg] : llvm::zip_equal(fieldNames, args)) {
276 B.create<WriteFieldOp>(loc, self, fieldName, arg);
277 }
278 // Return self
279 B.create<mlir::func::ReturnOp>(loc, mlir::ValueRange({self}));
280 }
281 );
282 });
283 auto compOp = cb.build(builder);
284 assert(compOp);
285}

299mlir::FailureOr<CtorCallBuilder> CtorCallBuilder::Make(
300 mlir::Operation *op, const zhl::TypeBinding &binding, mlir::OpBuilder &builder,
301 mlir::Value self, const zhl::TypeBindings &bindings
302) {
303 auto rootModule = op->getParentOfType<mlir::ModuleOp>();
304 auto *calleeComp = findCallee(binding.getName(), rootModule);
305 if (!calleeComp) {
306 return op->emitError() << "could not find component with name " << binding.getName();
307 }
308 auto constructorType = materializeTypeBindingConstructor(builder, binding, bindings);
309
310 return CtorCallBuilder(
311 constructorType, binding, op->getParentOfType<ComponentInterface>(), self,
312 calleeIsBuiltin(calleeComp)
313 );
314}

316mlir::Value CtorCallBuilder::build(
317 mlir::OpBuilder &builder, mlir::Location loc, mlir::ValueRange args,
318 GlobalBuilder::AndName globalBuilder
319) {
320 auto genericParams = compBinding.getGenericParamsMapping();
321 auto ref = builder.create<ConstructorRefOp>(
322 loc, compBinding.getName(), genericParams.size() - genericParams.sizeOfDeclared(), ctorType,
323 isBuiltin
324 );
325
326 auto call = builder.create<mlir::func::CallIndirectOp>(loc, ref, args);
327 Value compValue = call.getResult(0);
328
329 if (FrameSlot *slot = compBinding.getSlot()) {
330 assert(!globalBuilder && "cannot be both field and global");
331 zhl::ComponentSlot *compSlot = mlir::cast<zhl::ComponentSlot>(slot);
332 TypeBinding compSlotBinding = compSlot->getBinding();
333
334 // Create the field
335 FlatSymbolRefAttr name = createSlot(compSlot, builder, callerComponentOp, loc);
336 Type slotType = materializeTypeBinding(builder.getContext(), compSlotBinding);
337
338 compValue = storeAndLoadSlot(*compSlot, compValue, name, slotType, loc, builder, self);
339 } else if (globalBuilder) {
340 SymbolRefAttr name = GlobalBuilder::buildDef(*globalBuilder, loc, compValue.getType());
341 builder.create<SetGlobalOp>(loc, name, compValue);
342 compValue = builder.create<GetGlobalOp>(loc, compValue.getType(), name);
343 }
344
345 builder.create<ConstrainCallOp>(loc, compValue, args);
346 return compValue;
347}

363mlir::FailureOr<Value> coerceToArray(TypedValue<ComponentType> v, OpBuilder &builder) {
364 auto arraySuper = v.getType().getFirstMatchingSuperType([](Type t) {
365 if (auto ct = mlir::dyn_cast_if_present<ComponentType>(t)) {
366 return ct.isConcreteArray();
367 }
368 return false;
369 });
370
371 if (failed(arraySuper)) {
372 return failure();
373 }
374
375 return builder.create<SuperCoerceOp>(v.getLoc(), *arraySuper, v).getResult();
