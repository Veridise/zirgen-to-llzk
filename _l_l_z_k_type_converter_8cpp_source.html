55static Type deduceArrayType(Attribute attr) {
56 LLVM_DEBUG(llvm::dbgs() << "deduceArrayType(" << attr << ")\n");
57 if (auto typeAttr = dyn_cast<TypeAttr>(attr)) {
58 return typeAttr.getValue();
59 }
60 llvm_unreachable("Failed to convert array type");
61 return nullptr;
62}

72static SmallVector<Attribute>
73convertParamAttrs(ArrayRef<Attribute> in, LLZKTypeConverter &converter) {
74 return llvm::map_to_vector(in, [&](Attribute attr) -> Attribute {
75 return llvm::TypeSwitch<Attribute, Attribute>(attr)
76 .Case([&converter](TypeAttr typeAttr) {
77 return TypeAttr::get(converter.convertType(typeAttr.getValue()));
78 })
79 .Case(convert<zml::ConstExprAttr>)
80 .Case(convert<zml::LiftedExprAttr>)
81 .Default(pass<Attribute>);
82 });
83}

85static Attribute getSizeAttr(Attribute attr) {
86 LLVM_DEBUG(llvm::dbgs() << "getSizeSym(" << attr << ")\n");
87 return llvm::TypeSwitch<Attribute, Attribute>(attr)
88 .Case(pass<SymbolRefAttr>)
89 .Case(pass<IntegerAttr>)
90 .Case(convert<zml::ConstExprAttr>)
91 .Case(convert<zml::LiftedExprAttr>)
92 .Default([](auto) {
93 llvm_unreachable("was expecting a symbol, number, or an affine expression");
94 return nullptr;
95 });
96}

98llzk::LLZKTypeConverter::LLZKTypeConverter(const ff::FieldData &Field)
99 : field(Field),
100 feltEquivalentTypes(
101 {"Val", "Add", "Sub", "Mul", "BitAnd", "Inv", "Isz", "InRange", "Neg", "Mod"}
102 ),
103 extValBuiltins({"ExtVal", "ExtAdd", "ExtSub", "ExtMul", "ExtInv", "MakeExt"}) {
104
105 addConversion([](Type t) { return t; });
106
107 // Conversions from ZML to LLZK
108
109 addConversion([&](zml::ComponentType t) -> Type {
110 auto convertedAttrs = convertParamAttrs(t.getParams(), *this);
111 return llzk::component::StructType::get(
112 t.getName(), ArrayAttr::get(t.getContext(), convertedAttrs)
113 );
114 });
115
116 addConversion([&](zml::ComponentType t) -> std::optional<Type> {
117 if (extValBuiltins.find(t.getName().getValue()) != extValBuiltins.end() && t.getBuiltin()) {
118 return createArrayRepr(t.getContext());
119 }
120 return std::nullopt;
121 });
122
123 addConversion([&](zml::ComponentType t) -> std::optional<Type> {
124 if (t.getName().getValue() != "Array") {
125 return std::nullopt;
126 }
127 LLVM_DEBUG(llvm::dbgs() << "addConversion: " << t << "\n");
128 assert(t.getParams().size() == 2);
129 auto typeAttr = t.getParams()[0];
130 auto sizeAttr = t.getParams()[1];
131
132 llvm::SmallVector<Attribute> dims({getSizeAttr(sizeAttr)});
133 auto inner = convertType(deduceArrayType(typeAttr));
134 if (auto innerArr = dyn_cast<ArrayType>(inner)) {
135 auto innerDims = innerArr.getDimensionSizes();
136 dims.insert(dims.end(), innerDims.begin(), innerDims.end());
137 inner = innerArr.getElementType();
138 }
139
140 return ArrayType::get(inner, dims);
141 });
142
143 addConversion([](zml::ComponentType t) -> std::optional<Type> {
144 if (t.getName().getValue() == "String") {
145 return llzk::string::StringType::get(t.getContext());
146 }
147 return std::nullopt;
148 });
149
150 addConversion([&](zml::ComponentType t) -> std::optional<Type> {
151 if (feltEquivalentTypes.find(t.getName().getValue()) != feltEquivalentTypes.end() &&
152 t.getBuiltin()) {
153 return FeltType::get(t.getContext());
154 }
155 return std::nullopt;
156 });
157
158 addConversion([&](zml::VarArgsType t) {
159 std::vector<int64_t> shape = {ShapedType::kDynamic};
160 return ArrayType::get(convertType(t.getInner()), shape);
161 });
162
163 addConversion([](zml::TypeVarType t) { return TypeVarType::get(t.getContext(), t.getName()); });
164
165 addSourceMaterialization(unrealizedCastMaterialization);
166 addSourceMaterialization(
167 [this](
168 OpBuilder &builder, Type type, ValueRange inputs, Location loc
169 ) -> std::optional<Value> {
170 if (inputs.size() != 1) {
171 return std::nullopt;
172 }
173
174 auto inputType = inputs[0].getType();
175 auto convertedTarget = convertType(type);
176 if (!convertedTarget) {
177 return std::nullopt;
178 }
179 if (inputType == convertedTarget) {
180 return unrealizedCastMaterialization(builder, type, inputs, loc);
181 }
182 if (llzk::typesUnify(inputType, convertedTarget)) {
183 auto unifyCast = builder.create<UnifiableCastOp>(loc, convertedTarget, inputs[0]);
184 return unrealizedCastMaterialization(builder, type, unifyCast.getResult(), loc);
185 }
186 return std::nullopt;
187 }
188 );
189 addTargetMaterialization(unrealizedCastMaterialization);
190 addTargetMaterialization(
191 [this](
192 OpBuilder &builder, Type type, ValueRange inputs, Location loc
193 ) -> std::optional<Value> {
194 if (inputs.size() != 1) {
195 return std::nullopt;
196 }
197
198 auto convertedInput = convertType(inputs[0].getType());
199 if (!convertedInput) {
200 return std::nullopt;
201 }
202 if (convertedInput == type) {
203 return unrealizedCastMaterialization(builder, type, inputs, loc);
204 }
205 if (llzk::typesUnify(convertedInput, type)) {
206 auto cast = unrealizedCastMaterialization(builder, convertedInput, inputs, loc);
207 if (cast.has_value()) {
208 return builder.create<UnifiableCastOp>(loc, type, *cast);
209 }
210 }
211 return std::nullopt;
212 }
213 );
214 addArgumentMaterialization(unrealizedCastMaterialization);
215}

230Values LLZKTypeConverter::wrapArrayValues(Value v, OpBuilder &builder) const {
231 assertIsValidRepr(v);
232 Values vals;
233 vals.reserve(field.degree);
234 for (uint64_t i = 0; i < field.degree; i++) {
235 auto idx = builder.create<arith::ConstantIndexOp>(v.getLoc(), i);
236 vals.push_back(
237 ValueWrap(builder.create<ReadArrayOp>(v.getLoc(), v, ValueRange(idx)), builder, *this)
238 );
239 }
240
241 return vals;
242}
