42template <typename Key> class OpBindingsMap : public OpBindings {
43public:
44 using map = mlir::DenseMap<Key, mlir::FailureOr<TypeBinding>>;
45
46 typename map::iterator begin() { return bindings.begin(); }
47 typename map::const_iterator begin() const { return bindings.begin(); }
48 typename map::iterator end() { return bindings.end(); }
49 typename map::const_iterator end() const { return bindings.end(); }
50
51 bool contains(mlir::Operation *op) const final { return containsKey(opToKey(op)); }
52
53 bool missingBindings() const override {
54 return std::any_of(begin(), end(), [](auto &pair) { return mlir::failed(pair.second); });
55 }
56 void print(llvm::raw_ostream &os) const final {
57 for (auto &[op, type] : bindings) {
58 printKey(op, os);
59 printBinding(os, type);
60 }
61 }
62 mlir::FailureOr<TypeBinding> add(mlir::Operation *op, mlir::FailureOr<TypeBinding> type) final {
63 validateOp(op);
64 return addType(opToKey(op), type);
65 }
66 const mlir::FailureOr<TypeBinding> &get(mlir::Operation *op) const final {
67 validateOp(op);
68 return getType(opToKey(op));
69 }
70
71 mlir::FailureOr<TypeBinding> addType(Key k, mlir::FailureOr<TypeBinding> type) {
72 bindings.insert({k, type});
73 return type;
74 }
75
76 const mlir::FailureOr<TypeBinding> &getType(Key k) const {
77 if (bindings.find(k) == end()) {
78 return failed;
79 }
80 return bindings.at(k);
81 }
82
83 bool containsKey(Key k) const { return bindings.find(k) != end(); }
84
85 friend ZhlOpBindings;
86
87protected:
88 virtual Key opToKey(mlir::Operation *) const = 0;
89 virtual void validateOp(mlir::Operation *) const = 0;
90 virtual void printKey(Key k, llvm::raw_ostream &os) const = 0;
91
92private:
93 mlir::FailureOr<TypeBinding> failed;
94 map bindings;
95};
