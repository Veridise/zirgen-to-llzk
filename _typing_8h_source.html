49template <typename Op> class OpTypingRule : public TypingRule {
50public:
51 using TypingRule::TypingRule;
52
53 mlir::LogicalResult match(mlir::Operation *op) const override {
54 if (auto o = mlir::dyn_cast<Op>(op)) {
55 return match(o);
56 }
57 return mlir::failure();
58 }
59
60 mlir::FailureOr<TypeBinding> typeCheck(
61 mlir::Operation *op, mlir::ArrayRef<TypeBinding> operands, Scope &scope,
62 mlir::ArrayRef<const Scope *> regionScopes
63 ) const override {
64 assert(op->getParentOfType<zirgen::Zhl::ComponentOp>() == scope.getOp());
65 if (auto o = mlir::dyn_cast<Op>(op)) {
66 return typeCheck(o, operands, scope, regionScopes);
67 }
68 return mlir::failure();
69 }
70
71 mlir::FailureOr<std::vector<TypeBinding>> bindRegionArguments(
72 mlir::ValueRange args, mlir::Operation *op, mlir::ArrayRef<TypeBinding> operands, Scope &scope
73 ) const override {
74 if (auto o = mlir::dyn_cast<Op>(op)) {
75 return bindRegionArguments(args, o, operands, scope);
76 }
77 return mlir::failure();
78 }
79
80 virtual mlir::LogicalResult match(Op op) const { return mlir::success(); }
81
82 virtual mlir::FailureOr<TypeBinding>
83 typeCheck(Op op, mlir::ArrayRef<TypeBinding> operands, Scope &scope, mlir::ArrayRef<const Scope *>)
84 const = 0;
85
86 virtual mlir::FailureOr<std::vector<TypeBinding>>
87 bindRegionArguments(mlir::ValueRange args, Op, mlir::ArrayRef<TypeBinding>, Scope &) const {
88 assert(args.size() == 0); // An op that has arguments must override this method
89 return std::vector<TypeBinding>();
90 }
91};
