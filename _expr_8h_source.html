349template <typename Expr> class TypedExprView : public ExprView {
350public:
352 const Expr *get() const override { return mlir::cast_if_present<Expr>(base()); }
353
355 const Expr &ref() const override {
356 auto *E = get();
357 assert(E);
358 return *E;
359 }
360
363 const Expr &operator*() const { return ref(); }
364
367 const Expr *operator->() const { return get(); }
368
369 static bool classof(const ExprView *view) {
370 return view && mlir::isa_and_present<Expr>(view->operator->());
371 }
372
374 virtual const detail::ExprBase *base() const = 0;
375};

378template <typename Expr> class TypedExprViewAdaptor : public TypedExprView<Expr> {
379public:
380 TypedExprViewAdaptor() : view(nullptr) {}

382 TypedExprViewAdaptor(const ExprView &View) : view(&View) {}
383
384 static bool classof(const ExprView *view) {
385 return view && mlir::isa_and_present<Expr>(view->get());
386 }
387
388 const detail::ExprBase *base() const override {
389 if (view) {
390 return view->get();
391 }
392 return nullptr;
393 }
394
395 operator bool() const { return view && *view; }
396
399 operator ConstExpr() const override { return view ? ConstExpr(*view) : ConstExpr(); }
400
401private:
402 const ExprView *view;
403};

457#define enable_expr_cast(T, F) \
458 template <> \
459 struct llvm::CastInfo<T, F> : public llvm::CastIsPossible<T, std::remove_const_t<F>> { \
460 static inline T doCast(const std::remove_const_t<F> &f) { return T(f); } \
461 static inline T castFailed() { return T(); } \
462 static inline T doCastIfPossible(const std::remove_const_t<F> &f) { \
463 if (!llvm::CastInfo<T, F>::isPossible(f)) { \
464 return castFailed(); \
465 } \
466 return doCast(f); \
467 } \
468 }
