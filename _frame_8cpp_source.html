83bool FrameSlot::belongsTo(const Frame &frame) const {
84 // If the given frame points to the same as my parent then the slot belongs to the frame.
85 if (frame.info.get() == parentFrame) {
86 return true;
87 }
88
89 // If the frame has a parent slot, then the slot belongs to the frame if the parent slot belongs
90 // to the frame.
91 if (auto *parentSlot = getParentSlot()) {
92 return parentSlot->belongsTo(frame);
93 }
94 // Otherwise the slot does not belong to the frame.
95 return false;
96}

204template <typename T>
205T traverseNestedArrayFramesT(
206 const FrameSlot *root, const FrameSlot *slot, const T &t,
207 std::function<T(const ArrayFrame &, const T &)> onArrayFrame
208) {
209 if (!slot) {
210 return t;
211 }
212 if (root != slot) {
213 if (auto arrayFrame = mlir::dyn_cast<ArrayFrame>(slot)) {
216 auto arrayT = onArrayFrame(*arrayFrame, t);
217 return traverseNestedArrayFramesT(root, slot->getParentSlot(), arrayT, onArrayFrame);
218 }
219 }
220 return traverseNestedArrayFramesT(root, slot->getParentSlot(), t, onArrayFrame);
221}
222
223static void traverseNestedArrayFrames(
224 const FrameSlot *root, const FrameSlot *slot,
225 std::function<void(const ArrayFrame &)> onArrayFrame
226) {
227 size_t dummy = 0;
228 traverseNestedArrayFramesT<size_t>(root, slot, dummy, [&](const ArrayFrame &frame, auto &) {
229 onArrayFrame(frame);
230 return 0;
231 });
232}

248mlir::SmallVector<mlir::Value> ComponentSlot::collectIVs() const {
249 mlir::SmallVector<mlir::Value> vec;
250 traverseNestedArrayFrames(this, this, [&](const ArrayFrame &frame) {
251 vec.insert(vec.begin(), frame.getInductionVar());
252 });
