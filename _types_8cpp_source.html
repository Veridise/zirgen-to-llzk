28bool isValidZMLType(mlir::Type type) {
29 return llvm::isa<TypeVarType>(type) || llvm::isa<ComponentType>(type) ||
30 (llvm::isa<VarArgsType>(type) && isValidZMLType(llvm::cast<VarArgsType>(type).getInner()));
31}

64mlir::LogicalResult ComponentType::verify(
65 llvm::function_ref<mlir::InFlightDiagnostic()> emitError, ::mlir::FlatSymbolRefAttr compName,
66 mlir::Type superType, ::llvm::ArrayRef<::mlir::Attribute> params, bool
67) {
68 if (!superType && compName.getValue() != "Component") {
69 return emitError() << "malformed IR: super type for " << compName << " cannot be null";
70 }
71 // TODO: Maybe add a check that ensures that only known builtins can have the flag set to true?
72 if (superType) {
73 if (!mlir::isa<ComponentType, TypeVarType>(superType)) {
74 return emitError() << "unexpected type " << superType;
75 }
76 }
77 std::vector<mlir::LogicalResult> results;
78 std::transform(
79 params.begin(), params.end(), std::back_inserter(results),
80 [&](mlir::Attribute attr) { return checkValidParam(emitError, attr); }
81 );
82
83 if (results.empty()) {
84 return mlir::success();
85 }
86 return mlir::success(std::all_of(results.begin(), results.end(), mlir::succeeded));
87}

96FailureOr<Type> ComponentType::getFirstMatchingSuperType(llvm::function_ref<bool(Type)> pred
97) const {
98 if (pred(*this)) {
99 return *this;
100 }
101 if (auto super = getSuperType()) {
102 if (auto superComp = mlir::dyn_cast<ComponentType>(super)) {
103 return superComp.getFirstMatchingSuperType(pred);
104 } else if (pred(super)) {
105 return super;
106 }
107 }
108 return failure();
109}

111static FailureOr<Attribute> getArrayProperty(ComponentType t, size_t idx) {
112 if (!t.isArray()) {

153static uint8_t priority(Attribute attr) {
154 return llvm::TypeSwitch<Attribute, uint8_t>(attr).Case([](IntegerAttr) { return 3; }
155 ).Case([](ConstExprAttr) {
156 return 2;
157 }).Case([](LiftedExprAttr) {
158 return 2;
159 }).Case([](SymbolRefAttr) {
160 return 1;
161 }).Default([](auto) { return 0; });
162}

164static bool checkSubtypeViaConcreteness(Attribute lhsAttr, Attribute rhsAttr) {
165 auto lhs = priority(lhsAttr);
166 auto rhs = priority(rhsAttr);
167 return lhs && rhs && lhs >= rhs;
168}

170static bool equivalentParam(Attribute lhs, Attribute rhs) {
171 if (lhs == rhs) {
172 return true;
173 }
174
175 auto lhsType = mlir::dyn_cast_if_present<TypeAttr>(lhs);
176 auto rhsType = mlir::dyn_cast_if_present<TypeAttr>(rhs);
177
178 if (lhsType && rhsType) {
179 return equivalentTypeAttr(lhsType, rhsType);
180 }
181
182 // Check that we are not in the situation where one side is a TypeAttr and the other isn't
183 if (!neitherAre<TypeAttr>(lhs, rhs)) {
184 return false;
185 }
186
187 return checkSubtypeViaConcreteness(lhs, rhs);
188}

192static bool subtypeViaParams(const ComponentType &lhs, ComponentType rhs) {
193 // If they have the same name and number of parameters
194 bool sameName = lhs.getName() == rhs.getName();
195 bool sameNParams = lhs.getParams().size() == rhs.getParams().size();
196
197 if (!(sameName && sameNParams)) {
198 return false;
199 }
200
201 for (auto [lhsParam, rhsParam] : llvm::zip_equal(lhs.getParams(), rhs.getParams())) {
202 if (!equivalentParam(lhsParam, rhsParam)) {
203 return false;
204 }
205 }
206 return true;
207}
208
209static bool subtypeViaParams(const ComponentType &lhs, Type rhs) {
210 if (auto rhsAsComp = mlir::dyn_cast_if_present<ComponentType>(rhs)) {
211 return subtypeViaParams(lhs, rhsAsComp);
212 }
213 return false;
214}

216bool ComponentType::subtypeOf(Type other) const {
217 if (*this == other || subtypeViaParams(*this, other)) {
218 return true;
219 }
220 if (auto super = getSuperTypeAsComp()) {
221 return super.subtypeOf(other);
222 }
223 return false;
224}
