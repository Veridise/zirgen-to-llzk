40static Operation *findOpFromValue(Value v) {
41 if (auto op = v.getDefiningOp()) {
42 return op;
43 }
44 auto region = v.getParentRegion();
45 if (!region) {
46 return nullptr;
47 }
48 return region->getParentOp();
49}

51void ValueBindings::emitRemarks() const {
52 for (auto &[val, binding] : *this) {
53 auto op = findOpFromValue(val);
54 auto valIsOp = val.getDefiningOp() != nullptr;
55 assert(op);
56 if (mlir::failed(binding)) {
57 if (valIsOp) {
58 op->emitError() << "type check failed";
59 } else {
60 op->emitError() << "type check for value " << val << " failed ";
61 }
62 } else {
63 if (valIsOp) {
64 op->emitRemark() << "type: " << *binding;
65 } else {
66 op->emitRemark() << "type for value " << val << ": " << *binding;
67 }
68 }
69 }
70}

124 auto forEach = [&](auto &map) {
125 for (auto &[_, binding] : map) {
126 if (binding->hasClosure()) {

129 }
130 };
131 forEach(values.bindings);
