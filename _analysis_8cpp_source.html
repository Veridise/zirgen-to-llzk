75class ConcreteImpl : public zhl::ZIRTypeAnalysis::Impl {
76public:
77 explicit ConcreteImpl(mlir::ModuleOp module, mlir::OpBuilder &builder)
78 : typeBindings(builder.getUnknownLoc()), failureRef(failure()), opBindings(nullptr) {
79 std::unordered_set<std::string_view> definedNames;
80 for (auto op : module.getOps<zirgen::Zhl::ComponentOp>()) {
81 definedNames.insert(op.getName());
82 }
83 zml::addBuiltinBindings(typeBindings, definedNames);
84 auto opBindingsResult = typeCheck(module, typeBindings, zhlTypingRules(typeBindings));
85 if (failed(opBindingsResult)) {
86 typeCheckingFailed = true;
87 } else {
88 opBindings = std::move(*opBindingsResult);
89 }
90 }
91
92 const mlir::FailureOr<TypeBinding> &getType(mlir::Operation *op) const final {
93 if (typeCheckingFailed) {
94 return failureRef;
95 }
96 return opBindings->get(op);
97 }
98
99 const mlir::FailureOr<TypeBinding> &getType(mlir::Value value) const final {
100 if (typeCheckingFailed) {
101 return failureRef;
102 }
103 return opBindings->getValue(value);
104 }
105
106 mlir::FailureOr<TypeBinding> getType(mlir::StringRef name) const final {
107 if (typeCheckingFailed) {
108 return failureRef;
109 }
110 return typeBindings.MaybeGet(name);
111 }
112
113 mlir::LogicalResult addType(mlir::Value value, const TypeBinding &binding) final {
114 if (typeCheckingFailed) {
115 return mlir::failure();
116 }
117
118 return opBindings->addValue(value, binding);
119 }
120
121 const TypeBindings &getBindings() const override { return typeBindings; }
122
123 void print(llvm::raw_ostream &os) const final {
124 if (!typeCheckingFailed) {
125 opBindings->print(os);
126 } else {
127 os << "type checking process failed and did not produce bindings";
128 }
129 }
130
131 operator mlir::LogicalResult() const final {
132 return typeCheckingFailed || failed(*opBindings) ? failure() : success();
133 }
134
135 mlir::ArrayRef<TypeBinding *> getClosures() final {
136 if (!cachedClosures.has_value()) {
137 cachedClosures = opBindings->getClosures();
138 }
139 return *cachedClosures;
140 }
141
142 void emitRemarks() const final { opBindings->emitRemarks(); }
143
144private:
145 TypeBindings typeBindings;
146 bool typeCheckingFailed = false;
147 // Because the methods return references to a FailureOr
148 mlir::FailureOr<TypeBinding> failureRef;
149 std::unique_ptr<ZhlOpBindings> opBindings;
150 std::optional<SmallVector<TypeBinding *>> cachedClosures;
151};

159ZIRTypeAnalysis::ZIRTypeAnalysis(mlir::Operation *op, mlir::AnalysisManager &am) : impl(nullptr) {
160 if (auto mod = mlir::dyn_cast<mlir::ModuleOp>(op)) {
161 mlir::OpBuilder builder(mod.getBodyRegion());
162 impl = makeImpl<ConcreteImpl>(mod, builder);
163 } else {
164 auto parent = am.getCachedParentAnalysis<ZIRTypeAnalysis>(op->getParentOp());
165 assert(parent.has_value());
166 impl = makeImpl<DelegatedImpl>(*parent);
167 }
168}
