33ComponentBuilder &selfConstructs(ComponentBuilder &builder, mlir::Type type) {
34 return builder.fillBody({type}, {type}, [type](mlir::ValueRange args, mlir::OpBuilder &bldr) {
35 mlir::Location loc = getLocationForBuiltin(bldr);
36 // Reference to self
37 auto self = bldr.create<SelfOp>(loc, type);
38 // Construct Component superType
39 mlir::FunctionType constructor =
40 bldr.getFunctionType({}, ComponentType::Component(bldr.getContext()));
41 auto ref = bldr.create<ConstructorRefOp>(
42 loc, mlir::SymbolRefAttr::get(bldr.getStringAttr("Component")), constructor,
43 /*isBuiltin=*/true
44 );
45 auto comp = bldr.create<mlir::func::CallIndirectOp>(loc, ref);
46 // Store the result
47 bldr.create<WriteFieldOp>(loc, self, "$super", comp.getResult(0));
48 auto super = bldr.create<ReadFieldOp>(loc, comp.getResultTypes().front(), self, "$super");
49 bldr.create<ConstrainCallOp>(loc, super, mlir::ValueRange());
50 // Return self
51 bldr.create<mlir::func::ReturnOp>(loc, mlir::ValueRange({args[0]}));
52 });
53}

56void addBinOpCommon(mlir::OpBuilder &builder, mlir::StringRef name, ComponentType superType) {
57 auto componentType = ComponentType::get(builder.getContext(), name, superType, true);
58
59 builtinCommon(ComponentBuilder()
60 .name(name)
61 .field("$super", superType)
62 .fillBody(
63 {superType, superType}, {componentType},
64 [&superType, &componentType](mlir::ValueRange args, mlir::OpBuilder &bldr) {
65 mlir::Location loc = getLocationForBuiltin(bldr);
66 // Reference to self
67 auto self = bldr.create<SelfOp>(loc, componentType);
68 // Do the computation
69 auto op = bldr.create<OpTy>(loc, superType, args[0], args[1]);
70 // Store the result
71 bldr.create<WriteFieldOp>(loc, self, "$super", op);
72 // Return self
73 bldr.create<mlir::func::ReturnOp>(loc, mlir::ValueRange({self}));
74 }
75 )
76 ).build(builder);

380void zml::addBuiltins(
381 mlir::OpBuilder &builder, const std::unordered_set<std::string_view> &definedNames
382) {
383 assert(definedNames.find("Component") == definedNames.end() && "Can't redefine Component type");
384 addComponent(builder);
385
386 assert(definedNames.find("Val") == definedNames.end() && "Can't redefine Val type");
387 assert(definedNames.find("ExtVal") == definedNames.end() && "Can't redefine ExtVal type");
388 assert(definedNames.find("String") == definedNames.end() && "Can't redefine String type");
389 assert(definedNames.find("Array") == definedNames.end() && "Can't redefine Array type");
390 addTrivial(builder, "Val");
391 addTrivial(builder, "ExtVal");
392 addTrivial(builder, "String");
393 addArrayComponent(builder);
394
395 MAYBE("NondetReg") { addNondetReg(builder); }
396 MAYBE("NondetExtReg") { addNondetExtReg(builder); }
397 MAYBE("MakeExt") { addMakeExt(builder); }
398 MAYBE("EqzExt") { addEqzExt(builder); }
399 MAYBE("InRange") { addInRange(builder); }
400 MAYBE("BitAnd") { addBinOp<BitAndOp>(builder, "BitAnd"); }
401 MAYBE("Add") { addBinOp<AddOp>(builder, "Add"); }
402 MAYBE("Sub") { addBinOp<SubOp>(builder, "Sub"); }
403 MAYBE("Mul") { addBinOp<MulOp>(builder, "Mul"); }
404 MAYBE("ExtAdd") { addExtBinOp<ExtAddOp>(builder, "ExtAdd"); }
405 MAYBE("ExtSub") { addExtBinOp<ExtSubOp>(builder, "ExtSub"); }
406 MAYBE("ExtMul") { addExtBinOp<ExtMulOp>(builder, "ExtMul"); }
407 MAYBE("Mod") { addBinOp<ModOp>(builder, "Mod"); }
408 MAYBE("Inv") { addUnaryOp<InvOp>(builder, "Inv"); }
409 MAYBE("ExtInv") { addExtUnaryOp<ExtInvOp>(builder, "ExtInv"); }
