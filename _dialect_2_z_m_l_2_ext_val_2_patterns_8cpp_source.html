21template <typename Op> class ExtValOpConversionPattern : public OpConversionPattern<Op> {
22public:
23 ExtValOpConversionPattern(const TypeConverter &tc, MLIRContext *ctx, const BaseConverter &Conv)
24 : OpConversionPattern<Op>(tc, ctx), conv(&Conv) {}
25
26 using OpAdaptor = typename OpConversionPattern<Op>::OpAdaptor;
27
28 LogicalResult
29 matchAndRewrite(Op op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter) const override {
30 auto val = getConverter().lowerOp(op, adaptor, rewriter);
31 rewriter.replaceOp(op, val);
32 return success();
33 }
34
35private:
36 const BaseConverter &getConverter() const { return *conv; }
37
38 const BaseConverter *conv;
39};

47class EqzExtOpLowering : public OpConversionPattern<EqzExtOp> {
48public:
49 EqzExtOpLowering(const TypeConverter &tc, MLIRContext *ctx, const BaseConverter &Conv)
50 : OpConversionPattern<EqzExtOp>(tc, ctx), conv(&Conv) {}
51
52 LogicalResult matchAndRewrite(EqzExtOp op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter)
53 const override {
54 auto &helper = getConverter().getTypeHelper();
55 auto a = helper.wrapArrayValues(adaptor.getExtval(), rewriter);
56
57 // I'm not completely sure what are the exact semantics of this builtin but from what I could
58 // gather it asserts that the values of the extended element are zero.
59 // What I don't know is if it also emits a constraint.
60
61 SmallVector<Value> iszs;
62 std::transform(a.begin(), a.end(), std::back_inserter(iszs), [&](Value v) -> Value {
63 return helper.createIszOp(v, rewriter);
64 });
65 auto mult = std::reduce(iszs.begin() + 1, iszs.end(), *iszs.begin(), [&](auto lhs, auto rhs) {
66 return helper.createAndOp(lhs, rhs, rewriter);
67 });
68 Twine msg = "failed assertion: extended field element is not equal to zero";
69 rewriter.replaceOp(op, helper.createAssertOp(mult, rewriter.getStringAttr(msg), rewriter));
70
71 return success();
72 }
73
74private:
75 const BaseConverter &getConverter() const { return *conv; }
76
77 const BaseConverter *conv;
78};
