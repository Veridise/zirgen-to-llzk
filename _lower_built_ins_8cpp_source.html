36mlir::FailureOr<ConstructorRefOp> getConstructorRef(func::CallIndirectOp op) {
37 auto constrRef = mlir::dyn_cast<ConstructorRefOp>(op.getCallee().getDefiningOp());
38 if (!constrRef) {
39 return mlir::failure();
40 }
41 return constrRef;
42}

44mlir::FailureOr<std::string> getCallIndirectName(func::CallIndirectOp op) {
45 auto constrRef = getConstructorRef(op);
46 if (mlir::failed(constrRef)) {
47 return mlir::failure();
48 }
49 return constrRef->getComponent().str();
50}

52bool isLegalCallIndirect(func::CallIndirectOp op) {
53 auto name = getCallIndirectName(op);
54 if (mlir::failed(name)) {
55 return true;
56 }
57 auto ref = getConstructorRef(op);
58 if (mlir::failed(ref)) {
59 return true;
60 }
61 auto found = BuiltInComponentNames.find(*name) != BuiltInComponentNames.end();
62 auto markedBuiltin = ref->getBuiltin();
63 // Is the constructor call from a component that is built-in?
64 return !(found && markedBuiltin);
65}
66
67bool isLegalConstructRefOp(ConstructorRefOp op) {
68 auto found = BuiltInComponentNames.find(op.getComponent().str()) != BuiltInComponentNames.end();
69 auto markedBuiltin = op.getBuiltin();
70
71 return !(found && markedBuiltin);
72}

74template <typename NewOp, const char *Name>
75class ReplaceConstructorCallWithBuiltIn : public OpConversionPattern<func::CallIndirectOp> {
76public:
77 using OpConversionPattern<func::CallIndirectOp>::OpConversionPattern;
78
79 LogicalResult matchAndRewrite(
80 func::CallIndirectOp op,
81 typename OpConversionPattern<func::CallIndirectOp>::OpAdaptor adaptor,
82 ConversionPatternRewriter &rewriter
83 ) const override {
84 auto name = getCallIndirectName(op);
85 if (mlir::failed(name)) {
86 return name;
87 }
88
89 // If it's not the one we are looking for just reject the match
90 if (*name != Name) {
91 return mlir::failure();
92 }
93
94 mlir::ValueRange args(
95 mlir::iterator_range(adaptor.getOperands().begin() + 1, adaptor.getOperands().end())
96 );
97 rewriter.replaceOpWithNewOp<NewOp>(op, op.getResult(0).getType(), args);
98 return mlir::success();
99 }
100};

102class RemoveConstructorRef : public OpConversionPattern<ConstructorRefOp> {
103public:
104 using OpConversionPattern<ConstructorRefOp>::OpConversionPattern;
105
106 LogicalResult matchAndRewrite(ConstructorRefOp op, OpAdaptor, ConversionPatternRewriter &rewriter)
107 const override {
108 if (isLegalConstructRefOp(op)) {
109 return mlir::failure();
110 }
111
112 rewriter.eraseOp(op);
113 return mlir::success();
114 }
115};

133class LowerBuiltInsPass : public LowerBuiltInsBase<LowerBuiltInsPass> {
134
135 void runOnOperation() override {
136 auto op = getOperation();
137
138 ZMLTypeConverter typeConverter;
139 mlir::MLIRContext *ctx = op->getContext();
140 mlir::RewritePatternSet patterns(ctx);
141
142 patterns.add<
143 BitAndPattern, AddPattern, SubPattern, MulPattern, InvPattern, IszPattern, NegPattern,
144 ModPattern, InRangePattern, RemoveConstructorRef, ExtAddPattern, ExtSubPattern,
145 ExtMulPattern, ExtInvPattern, MakeExtPattern>(typeConverter, ctx);
146
147 // Set conversion target
148 mlir::ConversionTarget target(*ctx);
149 target.addLegalDialect<
150 ZMLDialect, mlir::func::FuncDialect, mlir::index::IndexDialect, mlir::scf::SCFDialect,
151 mlir::arith::ArithDialect>();
152 target.addLegalOp<mlir::UnrealizedConversionCastOp, mlir::ModuleOp>();
153
154 target.addDynamicallyLegalOp<func::CallIndirectOp>(isLegalCallIndirect);
155 target.addDynamicallyLegalOp<ConstructorRefOp>(isLegalConstructRefOp);
156
157 // Call partialTransformation
158 if (mlir::failed(mlir::applyFullConversion(op, target, std::move(patterns)))) {
159 signalPassFailure();
160 }
161 }
162};
