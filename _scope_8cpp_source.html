26LogicalResult Scope::declareGlobal(StringRef name, TypeBinding type, EmitErrorFn emitErr) {
27 MembersMap::iterator existing = globals.find(name);
28 // If there's an existing TypeBinding, the new one must be the same.
29 if (existing != globals.end()) {
30 if (std::optional<TypeBinding> &ev = existing->getValue()) {
31 if (ev.value() != type) {
32 return emitErr() << "Attempt to change type of global \"" << name << "\"\n"
33 << " old type = " << ev.value() << "\n"
34 << " new type = " << type << "\n";
35 }
36 return success();
37 }
38 }
39 globals[name] = type;
40 return success();
41}

52TypeBinding ComponentScope::declareLiftedAffineToGenericParam(const TypeBinding &type) {
53 assert(type.hasConstExpr() && "Can't lift a non constant binding");
54 assert(
55 mlir::isa<expr::CtorExpr>(type.getConstExpr()) &&
56 "Can't lift a constant binding that is not a constructor expression"
57 );
58
59 auto size = liftedParams.size();
60 auto allocName = ("Aff$" + Twine(size)).str();
61 auto paramBinding = TypeBinding::MakeGenericParam(bindings->Manage(type), allocName);
62 liftedParams.declare(allocName, paramBinding, liftedParams.size(), true);
63 return paramBinding;
64}

91TypeBinding ComponentScope::createBinding(StringRef name, Location loc) const {
92 assert(succeeded(superType));
93 auto nGenericParams = genericParams.size();
94 auto genericParamsCopy = genericParams;
95 for (auto &[paramName, paramData] : liftedParams) {
96 assert(
97 !genericParams.contains(paramName) &&
98 "A normal param's name collides with the autogenerated name"
99 );
100 genericParamsCopy.declare(
101 paramName, paramData.Type, paramData.Pos + nGenericParams, paramData.Injected
102 );
103 }
104 TypeBinding::Flags flags;
105 flags.setExtern(extern_).setBackVariablesNeed(needsBVs);
106 return bindings->Create(
107 name, loc, bindings->Manage(*superType), genericParamsCopy, constructorParams, members, flags,
108 frame
109 );
110}

176void collectTypeVarsFromGenericParams(
177 const TypeBinding &binding, llvm::StringMap<const TypeBinding *> &varNames
178) {
179 for (auto &param : binding.getGenericParams()) {
180 if (param.isGenericParam()) {
181 varNames.insert({param.getGenericParamName(), &param});
182 } else {
183 collectTypeVarsFromGenericParams(param, varNames);
184 }
185 }
186}

188TypeBinding BlockScope::createBinding(StringRef name, Location loc) const {
189 assert(succeeded(superType));
190 ParamsMap ctorArgs;
191 ctorArgs.declare("super", *superType, 0);
192 std::vector<std::string_view> sortedFieldNames;
193 sortedFieldNames.reserve(members.size());
194 llvm::StringMap<const TypeBinding *> varNames;
195
196 std::transform(
197 members.begin(), members.end(), std::back_inserter(sortedFieldNames),
198 [&](auto &p) {
199 collectTypeVarsFromGenericParams(*p.second, varNames);
200 return p.first();
201 }
202 );
203 std::sort(sortedFieldNames.begin(), sortedFieldNames.end());
204 size_t argNo = 1;
205 for (auto &fieldName : sortedFieldNames) {
206 auto memberBinding = members.at(fieldName);
207 assert(memberBinding.has_value());
208 ctorArgs.declare(fieldName, *memberBinding, argNo);
209 argNo++;
210 }
211 collectTypeVarsFromGenericParams(*superType, varNames);
212 ParamsMap genericParams;
213 size_t paramNo = 0;
214 for (auto &[varName, type] : varNames) {
215 genericParams.declare(varName, *type, paramNo);
216 paramNo++;
217 }
218
219 return TypeBinding::WithClosure(bindings->CreateAnon(
220 name, loc, bindings->Manage(*superType), genericParams, ctorArgs, members, getCurrentFrame()
221 ));
222}

231void LexicalScopeImpl::declareMemberImpl(StringRef name, TypeBinding type) {
232 // If the value is present and we are (re-)declaring it
233 // we can only do so if it has not value.
234 if (!memberDeclaredWithTypeImpl(name)) {
235 members[name] = type;
236 }
