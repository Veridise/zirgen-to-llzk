27template <int ArgN, typename Parent = mlir::func::FuncOp> class Arg {
28public:
29 template <typename Op>
30 static mlir::LogicalResult generate(
31 Op op, mlir::ConversionPatternRewriter &, const mlir::TypeConverter *,
32 mlir::SmallVector<mlir::Value, 1> &values
33 ) {
34 auto func = op->template getParentOfType<Parent>();
35 if (func.getNumArguments() <= ArgN) {
36 return mlir::failure();
37 }
38 values.push_back(func.getArgument(ArgN));
39 return mlir::success();
40 }
41};

46template <typename Op> class NewOp {
47public:
48 template <typename InOp>
49 static mlir::LogicalResult generate(
50 InOp op, mlir::ConversionPatternRewriter &rewriter, const mlir::TypeConverter *typeConverter,
51 mlir::SmallVector<mlir::Value, 1> &values
52 ) {
53 mlir::SmallVector<mlir::Type, 1> convertedTypes;
54 if (mlir::failed(typeConverter->convertTypes(op->getResultTypes(), convertedTypes))) {
55 return mlir::failure();
56 }
57 auto newOp = rewriter.create<Op>(op.getLoc(), convertedTypes, mlir::ValueRange());
58 values.push_back(newOp);
59 return mlir::success();
60 }
61};

66template <typename Strategy> class ReplaceSelfWith : public mlir::OpConversionPattern<SelfOp> {
67public:
68 using mlir::OpConversionPattern<SelfOp>::OpConversionPattern;
69
70 mlir::LogicalResult matchAndRewrite(
71 SelfOp op, typename mlir::OpConversionPattern<SelfOp>::OpAdaptor adaptor,
72 mlir::ConversionPatternRewriter &rewriter
73 ) const override {
74 mlir::SmallVector<mlir::Value, 1> selfReplacement;
75 if (mlir::failed(Strategy::generate(op, rewriter, getTypeConverter(), selfReplacement))) {
76 return mlir::failure();
77 }
78 rewriter.inlineBlockBefore(&op.getRegion().front(), op, selfReplacement);
79 rewriter.replaceOp(op, selfReplacement);
80 return mlir::success();
81 }
82};
