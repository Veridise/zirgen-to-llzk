56void ComponentBuilder::Ctx::checkBareRequirements() {
57 assert(loc.has_value());
58 assert(!compName.empty());
59 assert((isBuiltin || isClosure) == compAttrs.empty());
60 // If the component is a builtin it cannot use back-variables.
61 if (isBuiltin) {
62 assert(!usesBackVariables);

66ComponentOp ComponentBuilder::Ctx::buildBare(mlir::OpBuilder &builder) {
67 auto builtin = builtinAttrs(builder);
68 SmallVector<mlir::NamedAttribute> closure;
69 mlir::ArrayRef<mlir::NamedAttribute> attrs;
70 if (isClosure) {
71 attrs = closure;
72 } else {
73 attrs = isBuiltin ? mlir::ArrayRef<mlir::NamedAttribute>(builtin) : compAttrs;
74 }
75
76 if (isGeneric()) {
77 SmallVector<StringRef> typeParamsRefs =
78 llvm::map_to_vector(typeParams, [](auto &s) { return s.str(); });
79 return builder.create<ComponentOp>(*loc, compName, typeParamsRefs, attrs, usesBackVariables);
80 } else {
81 return builder.create<ComponentOp>(*loc, compName, attrs, usesBackVariables);
82 }
83}

87ComponentOp ComponentBuilder::build(mlir::OpBuilder &builder) {
88 if (!ctx.loc.has_value()) {
89 ctx.loc = builder.getUnknownLoc();
90 }
91 ctx.checkBareRequirements();
92 auto op = ctx.buildBare(builder);
93 if (ctx.deferCb) {
94 ctx.deferCb(op);
95 }
96
97 ctx.checkRequirements();
98
99 mlir::OpBuilder::InsertionGuard guard(builder);
100 auto block = builder.createBlock(&op.getRegion());
101 builder.setInsertionPointToStart(block);
102 ctx.addFields(builder);
103 ctx.addBody(op, builder);
104
105 ctx = Ctx();
106 return op;
107}

192void ComponentBuilder::FillBody::set(ComponentOp op, Ctx &buildCtx, mlir::OpBuilder &builder)
193 const {
194 buildCtx.constructorType = builder.getFunctionType(argTypes, results);
195 SmallVector<mlir::NamedAttribute> attrs = buildCtx.funcBodyAttrs(builder);
196
197 auto bodyOp = builder.create<mlir::func::FuncOp>(
198 builder.getUnknownLoc(), op.getBodyFuncName(), buildCtx.constructorType, attrs
199 );
200 mlir::OpBuilder::InsertionGuard insertionGuard(builder);
201 builder.setInsertionPointToStart(bodyOp.addEntryBlock());
202 delegate(bodyOp.getArguments(), builder);
203}

211void ComponentBuilder::TakeRegion::set(ComponentOp op, Ctx &buildCtx, mlir::OpBuilder &builder)
212 const {
213 auto constructorAttrs = buildCtx.funcBodyAttrs(builder);
214
215 assert(buildCtx.constructorType);
216 assert(body);
217 auto bodyOp = builder.create<mlir::func::FuncOp>(
218 *buildCtx.loc, op.getBodyFuncName(), buildCtx.constructorType, constructorAttrs
219 );
220
221 // Create arguments for the entry block (aka region arguments)
222 auto &entryBlock = bodyOp.getRegion().emplaceBlock();
223 assert(bodyOp.getRegion().hasOneBlock());
224 entryBlock.addArguments(buildCtx.constructorType.getInputs(), buildCtx.argLocs);
225
226 mlir::OpBuilder::InsertionGuard guard(builder);
227 builder.setInsertionPointToStart(&entryBlock);
228
229 auto self = builder.create<zml::SelfOp>(op.getLoc(), op.getType(), *body);
230 builder.create<mlir::func::ReturnOp>(self.getLoc(), mlir::ValueRange({self}));
231}
