46static bool isTransitivelyValImpl(const TypeBinding &type) {
47 const auto *t = &type;
48 while (t->hasSuperType()) {
49 t = &t->getSuperType();
50 if (t->getName() == "Val") {
51 return true;
52 }
53 }
54 return false;
55}

81void TypeBinding::setName(StringRef newName) {
82 // If the binding has a slot check if it can be converted to a ComponentSlot,
83 // assert that the type is equal (before the change of name)
84 // and then rename the inner binding in the slot.
85 if (auto *compSlot = mlir::dyn_cast_if_present<ComponentSlot>(slot)) {
86 assert(compSlot->contains(*this));
87 compSlot->editInnerBinding([&](TypeBinding &inner) {
88 if (this != &inner) {
89 inner.setName(newName);
90 }
91 });
92 }
93 name = newName;
94}

96mlir::LogicalResult TypeBinding::subtypeOf(const TypeBinding &other) const {
97 if (name == BOTTOM) {
98 return mlir::success();
99 }
100 if (*this == other) {
101 return mlir::success();
102 }
103 // Special case for arrays
104 if (name == "Array" && other.name == "Array") {
105 auto &arrElt = getGenericParamsMapping().getParam(0);
106 auto &otherArrElt = other.getGenericParamsMapping().getParam(0);
107 auto &arrSize = getGenericParamsMapping().getParam(1);
108 auto &otherArrSize = other.getGenericParamsMapping().getParam(1);
109 if (arrSize == otherArrSize) {
110 return arrElt.subtypeOf(otherArrElt);
111 }
112 return failure();
113 }
114
115 if (superType != nullptr) {
116 return superType->subtypeOf(other);
117 }
118 return mlir::failure();
119}

121TypeBinding TypeBinding::commonSupertypeWith(const TypeBinding &other) const {
122// Wrapped this way instead of LLVM_DEBUG because the macro may create its own scope.
123#ifndef NDEBUG
124 static unsigned indentLevel = 0;
125 indentLevel++;
126 deferDecr defer{.value = indentLevel};
127 auto logLine = [&](unsigned offset = 0) -> llvm::raw_ostream & {
128 return llvm::dbgs().indent(indentLevel + offset);
129 };
130 auto nl = []() { llvm::dbgs() << '\n'; };
131#endif
132 LLVM_DEBUG(print(logLine() << " this = ", true); nl();
133 other.print(logLine() << " other = ", true); nl(););
134 if (*this == other) {
135 LLVM_DEBUG(logLine(1) << "They are equal\n");
136 return *this;
137 }
138 // Special cases for arrays:
139 if (name == "Array") {
140 // If they are the same size the common super type is another array
141 // whose element is the least common type of the two element types
142 if (other.name == "Array") {
143 auto &arrElt = getGenericParamsMapping().getParam(0);
144 LLVM_DEBUG(arrElt.print(logLine(1) << "This array element: ", true); nl());
145 auto &otherArrElt = other.getGenericParamsMapping().getParam(0);
146 LLVM_DEBUG(otherArrElt.print(logLine(1) << "Other array element: ", true); nl());
147 auto &arrSize = getGenericParamsMapping().getParam(1);
148 LLVM_DEBUG(arrSize.print(logLine(1) << "This array size: ", true); nl());
149 auto &otherArrSize = other.getGenericParamsMapping().getParam(1);
150 LLVM_DEBUG(otherArrSize.print(logLine(1) << "Other array size: ", true); nl());
151 if (arrSize == otherArrSize) {
152 auto commonInner = arrElt.commonSupertypeWith(otherArrElt);
153 auto copy = *this;
154 copy.getGenericParamsMapping().getParam(0) = commonInner;
155 copy.selfConstructs();
156 LLVM_DEBUG(copy.print(logLine(1) << "Resolved a new array: ", true); nl());
157 return copy;
158 }
159 LLVM_DEBUG(logLine(1) << "Defaulting to Component\n");
160 return TypeBinding(loc);
161 }
162 // If this is an Array and the other has an Array supertype get the Array super type and go from
163 // there.
164 auto otherArray = other.getConcreteArrayType();
165 if (succeeded(otherArray)) {
166 LLVM_DEBUG(otherArray->print(logLine(1) << "Jumped to Array super type: ", true); nl());
167 return commonSupertypeWith(*otherArray);
168 }
169 }
170 if (mlir::succeeded(subtypeOf(other))) {
171 LLVM_DEBUG(logLine(1) << "this is a subtype of other\n");
172 return other;
173 }
174 if (mlir::succeeded(other.subtypeOf(*this))) {
175 LLVM_DEBUG(logLine(1) << "other is a subtype of this\n");
176 return *this;
177 }
178
179 if (superType) {
180 // Climb alternatively
181 LLVM_DEBUG(superType->print(logLine(1) << "Delegating to super type: ", true); nl());
182 return other.commonSupertypeWith(*superType);
183 }
184 LLVM_DEBUG(logLine(1) << "Defaulting to Component\n");
185 return TypeBinding(loc);
186}

206void TypeBinding::markSlot(FrameSlot *newSlot) {
207 if (slot == newSlot) {
208 return;
209 }
210 LLVM_DEBUG(print(llvm::dbgs() << "Binding "); llvm::dbgs() << ": Current slot is ";
211 p(slot) << " and the new slot is "; p(newSlot) << '\n');
212 assert((!slot == !!newSlot) && "Writing over an existing slot!");
213 slot = newSlot;

225FailureOr<TypeBinding> TypeBinding::getMember(StringRef memberName, EmitErrorFn emitError) const {
226 auto memberLookupRes = locateMember(memberName);
227 if (failed(memberLookupRes)) {
228 return emitError() << "member " << getName() << "." << memberName << " was not found";
229 }
230 // We check this after ensuring the member exists to give a more accurate error message
231 if (isPrivate(memberName)) {
232 return emitError() << "member " << getName() << "." << memberName
233 << " is private and cannot be accessed";
234 }
235 if (!memberLookupRes->has_value()) {
236 return emitError() << "internal error: could not deduce the type of member " << getName() << "."
237 << memberName;
238 }
239 return **memberLookupRes;
240}

242FailureOr<TypeBinding> TypeBinding::map(Params params, EmitErrorFn emitError) const {
243 if (isGenericParam() && params.contains(getGenericParamName())) {
244 auto result = params[getGenericParamName()];
245 if (!result) {
246 return emitError() << "parameter '" << getGenericParamName() << "' not found";
247 }
248 return *result;
249 }
250 if (hasConstExpr()) {
251 auto replacement = constExpr.remap(params, emitError);
252 if (failed(replacement)) {
253 return failure();
254 }
255 return WithExpr(*this, *replacement);
256 }
257 return *this;
258}

268static mlir::FailureOr<TypeBinding>
269getArrayProperty(size_t idx, EmitErrorFn emitError, const TypeBinding &binding) {
270 auto name = binding.getName();
271 auto params = binding.getGenericParamsMapping();
272 if (!binding.isArray()) {
273 return emitError() << "non array type '" << name << "' cannot be subscripted";
274 }
275 if (name == "Array") {
276 assert(params.size() == 2);
277 return params.getParam(idx);
278 }
279 // A component with an array super type can behave like an array but it doesn't have all the
280 // information required. In that case we defer the answer to the super type.
281 if (!binding.hasSuperType()) {
282 return failure();
283 }
284 // Get the property from the parent and adapt it to this context.
285 return maybeMap(getArrayProperty(idx, emitError, binding.getSuperType()), params, emitError);
286}

362void TypeBinding::print(llvm::raw_ostream &os, bool fullPrintout) const {
363 auto printType = [&]() {
364 os << name.ref();
365 if (flags.isSpecialized()) {
366 getGenericParamsMapping().printParams(os, {.fullPrintout = false});
367 } else {
368 getGenericParamsMapping().printNames(os);
369 }
370 if (fullPrintout) {
371 getConstructorParams().printParams(
372 os, {.fullPrintout = false, .printIfEmpty = true, .header = '(', .footer = ')'}
373 );
374 }
375 if (flags.isVariadic()) {
376 os << "...";
377 }
378 };
379 if (isConst()) {
380 if (hasConstValue(constExpr)) {
381 os << getConstValue(constExpr);
382 } else {
383 os << "?";
384 }
385 if (fullPrintout) {
386 os << " : ";
387 printType();
388 }
389 } else if (hasConstExpr()) {
390 getConstExpr()->print(os);
391 if (fullPrintout) {
392 os << " : ";
393 printType();
394 }
395 } else if (isGenericParam()) {
396 os << *genericParamName;
397 if (fullPrintout) {
398 os << " : ";
399 printType();
400 }
401 } else {
402 printType();
403 }
404 if (fullPrintout) {
405 os << " { ";
406 if (flags.isVariadic()) {
407 os << "variadic ";
408 }
409 if (flags.isSpecialized()) {
410 os << "specialized ";
411 }
412 if (flags.isSelfConstructor()) {
413 os << "selfConstructor ";
414 }
415 if (flags.isBuiltin()) {
416 os << "builtin ";
417 }
418 if (flags.isClosure()) {
419 os << "closure ";
420 }
421 if (flags.isExtern()) {
422 os << "extern ";
423 }
424 if (hasConstValue(constExpr)) {
425 os << "const(" << getConstValue(constExpr) << ") ";
426 }
427 if (genericParamName.has_value()) {
428 os << "genericParam(" << *genericParamName << ") ";
429 }
430 os << "constExpr(";
431 if (constExpr) {
432 constExpr->print(os);
433 } else {
434 os << "<<NULL>>";
435 }
436 os << ") ";
437 os << "}";
438 if (!members.empty()) {
439 os << " members { ";
440 llvm::interleaveComma(members, os, [&](auto &member) {
441 os << member.getKey() << ": ";
442 auto &type = member.getValue();
443 if (type.has_value()) {
444 type->print(os, false);
445 } else {
446 os << "⊥";
447 }
448 });
449 os << " }";
450 }
451 if (hasSuperType()) {
452 os << " :> ";
453 getSuperType().print(os, fullPrintout);
454 }
455 }
456}

466bool TypeBinding::operator==(const TypeBinding &other) const {
467 bool superTypeIsEqual;
468 if (superType == nullptr && other.superType == nullptr) {
469 superTypeIsEqual = true;
470 } else if (superType == nullptr || other.superType == nullptr) {
471 superTypeIsEqual = false;
472 } else {
473 superTypeIsEqual = *superType == *other.superType;
474 }
475
476 return superTypeIsEqual && flags == other.flags && name == other.name &&
477 constExpr == other.constExpr && genericParamName == other.genericParamName &&
478 members == other.members && getGenericParamsMapping() == other.getGenericParamsMapping() &&
479 getConstructorParams() == other.getConstructorParams();
480}

482FailureOr<std::optional<TypeBinding>> TypeBinding::locateMember(StringRef memberName) const {
483 auto it = members.find(memberName);
484 if (it != members.end()) {
485 return it->second;
486 }
487 if (superType != nullptr) {
488 return superType->locateMember(memberName);
489 }
490 return failure();
491}
