26ParamsStorage::ParamsStorage(
27 const ParamsMap &map, size_t size, std::optional<TypeBinding> defaultBinding
28)
29 : names(size),
30 injected((
31 assert(size <= std::numeric_limits<unsigned int>::max()), static_cast<unsigned int>(size)
32 )) {
33 // Hack to get the bindings ordered without having a default constructor
34 SmallVector<const TypeBinding *> tmp(size, nullptr);
35 for (auto &entry : map) {
36 const ParamData &val = entry.getValue();
37 uint64_t pos = val.Pos;
38 tmp[pos] = &val.Type;
39 names[pos] = entry.getKey();
40 assert(pos <= std::numeric_limits<unsigned int>::max());
41 injected[static_cast<unsigned int>(pos)] = val.Injected;
42 }
43 params.reserve(size);
44 for (const TypeBinding *type : tmp) {
45 if (type) {
46 params.push_back(*type);
47 } else {
48 assert(defaultBinding.has_value() && "cannot have null type when there is no default");
49 params.push_back(defaultBinding.value());
50 }
51 }
52}

58Params::operator ParamsMap() const {
59 ParamsMap map;
60 for (size_t i = 0; i < data()->params.size(); i++) {
61 assert(i <= std::numeric_limits<unsigned int>::max());
62 bool injected = data()->injected[static_cast<unsigned int>(i)];
63 map.declare(data()->names[i], data()->params[i], i, injected);
64 }
65 return map;
66}

108template <typename It, typename UnaryFn, typename Side>
109static void printHelper(It it, llvm::raw_ostream &os, UnaryFn fn, Side header, Side footer) {
110 os << header;
111 llvm::interleaveComma(it, os, fn);
112 os << footer;

115void Params::printMapping(llvm::raw_ostream &os, ParamsPrintCfg cfg) const {
116 printHelper(llvm::zip_equal(data()->names, data()->params), os, [&](auto tup) {
