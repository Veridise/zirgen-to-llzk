40class Materializer {
41public:
42 explicit Materializer(MLIRContext *ctx) : context(ctx) {}

44 Type materializeTypeBinding(const TypeBinding &binding) {
45 seen.clear();
46 return materializeImpl(binding);
47 }
48
49private:
50 Type materializeImpl(const TypeBinding &binding) {
51 checkCycle(binding);
52
53 auto base = materializeBaseType(binding);
54
55 seen.pop_back();
56 if (binding.isVariadic()) {
57 return VarArgsType::get(context, base);
58 }
59 return base;
60 }

62 SmallVector<Attribute> materializeGenericParamNames(const TypeBinding &binding) {
63 auto names = binding.getGenericParamNames();
64 return llvm::map_to_vector(names, [&](const auto &name) -> Attribute {
65 return SymbolRefAttr::get(StringAttr::get(context, name));
66 });
67 }

69 Attribute materializeAttribute(const TypeBinding &binding) {
70 // Special case for constants that do not have a known value
71 if (binding.isConst() && !binding.isKnownConst()) {
72 auto intAttr = IntegerAttr::get(IntegerType::get(context, 64), ShapedType::kDynamic);
73 LLVM_DEBUG(
74 llvm::dbgs() << "<== Materializing " << binding << " to IntegerAttr: " << intAttr << "\n"
75 );
76 return intAttr;
77 }
78 if (binding.isGenericParam() && !binding.getSuperType().isTypeMarker()) {
79 Attribute symAttr =
80 FlatSymbolRefAttr::get(StringAttr::get(context, binding.getGenericParamName()));
81 if (binding.getSuperType().hasConstExpr()) {
82 if (auto constExprAttr =
83 mlir::dyn_cast<ConstExprAttr>(materializeAttribute(binding.getSuperType()))) {
84 symAttr = LiftedExprAttr::get(mlir::dyn_cast<FlatSymbolRefAttr>(symAttr), constExprAttr);
85 }
86 }
87
88 LLVM_DEBUG(
89 llvm::dbgs() << "<== Materializing " << binding << " to FlatSymbolRefAttr: " << symAttr
90 << "\n"
91 );
92 return symAttr;
93 }
94 if (binding.hasConstExpr()) {
95 Builder builder(context);
96 auto attrResult = binding.getConstExpr().convertIntoAttribute(builder);
97 LLVM_DEBUG(
98 llvm::dbgs() << "<== Materializing " << binding << " to Attribute: " << attrResult << "\n"
99 );
100 return attrResult;
101 }
102
103 auto typeAttr = TypeAttr::get(materializeImpl(binding));
104 LLVM_DEBUG(
105 llvm::dbgs() << "<== Materializing " << binding << " to TypeAttr: " << typeAttr << "\n"
106 );
107 return typeAttr;
108 }

110 Type materializeGenericType(const TypeBinding &binding, Type superType) {
111 if (!binding.isSpecialized()) {
112 auto params = materializeGenericParamNames(binding);
113 return ComponentType::get(context, binding.getName(), superType, params, binding.isBuiltin());
114 }
115
116 auto paramBindings = binding.getGenericParams();
117 SmallVector<Attribute, 2> params =
118 map_to_vector(paramBindings, std::bind_front(&Materializer::materializeAttribute, this));
119
120 return ComponentType::get(context, binding.getName(), superType, params, binding.isBuiltin());
121 }
122
123 Type materializeBaseType(const TypeBinding &binding) {
124 if (!binding.hasSuperType() || binding.isTypeMarker()) {
125 return ComponentType::Component(context);
126 }
127 if (binding.isConst()) {
128 return ComponentType::Val(context);
129 }
130
131 if (binding.isGenericParam()) {
132 if (binding.getSuperType().isTypeMarker()) {
133 LLVM_DEBUG(llvm::dbgs() << "<== Materializing " << binding << " to TypeVarType\n");
134 return TypeVarType::get(
135 context, SymbolRefAttr::get(StringAttr::get(context, binding.getGenericParamName()))
136 );
137 }
138 if (binding.getSuperType().isVal() || binding.getSuperType().isTransitivelyVal()) {
139 LLVM_DEBUG(llvm::dbgs() << "<== Materializing " << binding << " to Val\n");
140 return ComponentType::Val(context);
141 }
142
143 assert(false && "Generic param that is neither Val or Type");
144 }
145 auto superType = materializeImpl(binding.getSuperType());
146 assertValidSuperType(superType);
147
148 if (binding.isGeneric()) {
149 return materializeGenericType(binding, superType);
150 }
151 return ComponentType::get(context, binding.getName(), superType, binding.isBuiltin());
152 }
153
154 void checkCycle(const TypeBinding &binding) {
155 std::string s;
156 llvm::raw_string_ostream ss(s);
157 binding.print(ss);
158 std::unordered_set set(seen.begin(), seen.end());
159 assert(set.find(s) == set.end() && "cycle detected");
160 seen.push_back(s);
161 }
162
163 MLIRContext *context;
164 std::vector<std::string> seen;
165};

180static Type specializeAndMaterializeTypeBinding(
181 MLIRContext *ctx, const TypeBinding &binding, const Params &scope, const TypeBindings &bindings
182) {
183 // If the scope is empty or the binding we are specializing is not generic then there are no type
184 // variables.
185 if (scope.empty() || !binding.isGeneric()) {
186 LLVM_DEBUG(llvm::dbgs() << binding << ": no specialization needed\n");
187 return materializeTypeBinding(ctx, binding);
188 }
189
190 // Make a copy to assign the params to
191 auto copy = binding;
192 ParamsScopeStack scopeStack(scope);
193
194 if (failed(zhl::specializeTypeBinding(&copy, scopeStack, bindings))) {
195 LLVM_DEBUG(llvm::dbgs() << "Failed to specialize binding " << binding << "\n");
196 return nullptr;
197 }
198
199 return materializeTypeBindingImpl(ctx, copy);
200}

212FunctionType zml::materializeTypeBindingConstructor(
213 OpBuilder &builder, const TypeBinding &binding, const TypeBindings &bindings
214) {
215 auto genericParams = binding.getGenericParamsMapping();
216 LLVM_DEBUG(llvm::dbgs() << "Materializing constructor type for " << binding << "\n");
217 // Create the type of the binding and of each argument
218 // then return a function type using the generated types.
219 // If any of the given types is a null just return nullptr for the whole thing.
220 std::vector<Type> args;
221 auto retType =
222 specializeAndMaterializeTypeBinding(builder.getContext(), binding, genericParams, bindings);
223 if (!retType) {
224 LLVM_DEBUG(indent() << "failed to materialize the return type for " << binding << "\n");
225 return nullptr;
226 }
227 LLVM_DEBUG(indent() << "Materialized return type to " << retType << "\n");
228
229 LLVM_DEBUG(indent() << "For binding " << binding << " constructor types are: \n");
230 auto params = binding.getConstructorParams();
231 std::transform(params.begin(), params.end(), std::back_inserter(args), [&](auto &argBinding) {
232 LLVM_DEBUG(indent(2) << argBinding << "\n");
233 auto materializedType = specializeAndMaterializeTypeBinding(
234 builder.getContext(), argBinding, genericParams, bindings
235 );
236 LLVM_DEBUG(indent() << "Materialized to " << materializedType << "\n");
237 return materializedType;
238 });
239 if (std::any_of(args.begin(), args.end(), [](Type t) { return !t; })) {
240 LLVM_DEBUG(indent() << "Failed to materialize an argument type for " << binding << "\n");
241 return nullptr;
242 }
243
244 auto ctorType = builder.getFunctionType(args, retType);
245 LLVM_DEBUG(
246 llvm::dbgs() << "<== Materializing constructor function of " << binding << " to " << ctorType
247 << "\n"
248 );
249 return ctorType;
250}
