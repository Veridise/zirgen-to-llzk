27/********************************************************************************************
28 *
29 * The grammar to represent semi-affine expressions with MLIR is as follows:
30 *
31 * semi-affine-expr ::= `(` semi-affine-expr `)`
32 * | semi-affine-expr `+` semi-affine-expr
33 * | semi-affine-expr `-` semi-affine-expr
34 * | symbol-or-const `*` semi-affine-expr
35 * | semi-affine-expr `ceildiv` symbol-or-const
36 * | semi-affine-expr `floordiv` symbol-or-const
37 * | semi-affine-expr `mod` symbol-or-const
38 * | bare-id
39 * | `-`? integer-literal
40 *
41 * symbol-or-const ::= `-`? integer-literal | symbol-id
42 *
43 * The interpreter function for TypeParamOp and LiteralOp are affine by definition, and the only
44 * place where we need to check if the expression is going to be affine is in the interpreter
45 * function for ConstructOp. Each component in the set below corresponds to an operation in the
46 * grammar above, with `Div` as a special case since it can map to two different division
47 * operations. In this implementation the `Div` component represents the `floordiv` affine operator.
48 *
49 * The definition of semi-affine is not completely correct with respect to MLIR's, a single
50 * symbol-id will be considered a semi-affine expression. However, when materializing the expression
51 * into an attribute a ConstExpr of type SymExpr will generate a FlatSymbolRefAttr and not an
52 * AffineMapAttr, which removes the corner case from the actual semi-affine expressions when
53 * generated.
54 *
55 ******************************************************************************************/
56
57// Set of components that could potentially be used to construct an affine expression
58static llvm::StringSet<> AffineRepresentableComponents({"Add", "Sub", "Mul", "Div", "Mod", "Neg"});
59
60inline bool isSemiAffine(const TypeBinding &binding) { return binding.hasConstExpr(); }
61
64inline bool isSymbolOrConst(const TypeBinding &binding) {
65 return isSemiAffine(binding) &&
66 (mlir::isa<expr::ValExpr, expr::SymExpr>(binding.getConstExpr()) ||
67 mlir::cast<expr::CtorExpr>(binding.getConstExpr())->getTypeName() == "Neg");
68}
69
70inline bool isIntLit(const TypeBinding &binding) {
71 return isSemiAffine(binding) && mlir::isa<expr::ValExpr>(binding.getConstExpr());
72}
73
74enum class Cmps : uint8_t { Add = 0, Sub, Mul, Div, Mod, Neg };
75
76FailureOr<Cmps> getToken(StringRef name) {
77 return llvm::StringSwitch<FailureOr<Cmps>>(name)
78 .Case("Add", Cmps::Add)
79 .Case("Sub", Cmps::Sub)
80 .Case("Mul", Cmps::Mul)
81 .Case("Div", Cmps::Div)
82 .Case("Mod", Cmps::Mod)
83 .Case("Neg", Cmps::Neg)
84 .Default(failure());
85}
86
87inline uint8_t arity(Cmps cmp) {
88 switch (cmp) {
89 case Cmps::Add:
90 case Cmps::Sub:
91 case Cmps::Mul:
92 case Cmps::Div:
93 case Cmps::Mod:
94 return 2;
95 case Cmps::Neg:
96 return 1;
97 }
98 llvm_unreachable("switch case with closed values");
99 return 0;
100}

102inline ConstExpr BinaryOp(StringRef name, ArrayRef<TypeBinding> args) {
103 return ConstExpr::Ctor(name, {args[0].getConstExpr(), args[1].getConstExpr()});
104}

106inline ConstExpr UnaryOp(StringRef name, ArrayRef<TypeBinding> args) {
107 return ConstExpr::Ctor(name, {args[0].getConstExpr()});
108}

114template <> TypeBinding interpretOp(zirgen::Zhl::TypeParamOp op, const TypeBinding &binding) {
115 if (binding.getSuperType().isVal()) {
116 return TypeBinding::WithExpr(binding, ConstExpr::Symbol(op.getName(), op.getIndex()));
117 }
118 return TypeBinding::NoExpr(binding);
119}

121template <> TypeBinding interpretOp(zirgen::Zhl::LiteralOp op, const TypeBinding &binding) {
122 // The LiteralOp typing rule must have already created a const. Here we just check that that is
123 // indeed the case.
124 assert(binding.isKnownConst() && binding.getConst() == op.getValue());
125 return binding;
126}

132template <>
133TypeBinding
134interpretOp(zirgen::Zhl::ConstructOp op, const TypeBinding &binding, ArrayRef<TypeBinding> &&args) {
135 LLVM_DEBUG(
136 llvm::dbgs() << "interpretating " << op << " at " << op.getLoc() << " with type " << binding
137 << "\n"
138 );
139 if (!AffineRepresentableComponents.contains(binding.getName())) {
140 LLVM_DEBUG(llvm::dbgs() << " is not affine representable\n");
141 return TypeBinding::NoExpr(binding);
142 }
143
144 auto tok = getToken(binding.getName());
145 assert(succeeded(tok));
146 assert(args.size() == arity(*tok));
147 switch (*tok) {
148 case Cmps::Add:
149 // semi-affine-expr `+` semi-affine-expr
150 if (isSemiAffine(args[0]) && isSemiAffine(args[1])) {
151 LLVM_DEBUG(llvm::dbgs() << " valid affine Add\n");
152 return TypeBinding::WithExpr(binding, BinaryOp("Add", args));
153 }
154 LLVM_DEBUG(llvm::dbgs() << " invalid affine Add\n");
155 break;
156 case Cmps::Sub:
157 // semi-affine-expr `-` semi-affine-expr
158 if (isSemiAffine(args[0]) && isSemiAffine(args[1])) {
159 LLVM_DEBUG(llvm::dbgs() << " valid affine Sub\n");
160 return TypeBinding::WithExpr(binding, BinaryOp("Sub", args));
161 }
162 LLVM_DEBUG(llvm::dbgs() << " invalid affine Sub\n");
163 break;
164 case Cmps::Mul:
165 // symbol-or-const `*` semi-affine-expr
166 if (isSymbolOrConst(args[0]) && isSemiAffine(args[1])) {
167 LLVM_DEBUG(llvm::dbgs() << " valid affine Mul\n");
168 return TypeBinding::WithExpr(binding, BinaryOp("Mul", args));
169 }
170 LLVM_DEBUG(llvm::dbgs() << " invalid affine Mul\n");
171 break;
172 case Cmps::Div:
173 // semi-affine-expr `floordiv` symbol-or-const
174 if (isSemiAffine(args[0]) && isSymbolOrConst(args[1])) {
175 LLVM_DEBUG(llvm::dbgs() << " valid affine Div\n");
176 return TypeBinding::WithExpr(binding, BinaryOp("Div", args));
177 }
178 LLVM_DEBUG(
179 llvm::dbgs() << " invalid affine Div | lhs = " << isSemiAffine(args[0])
180 << ", rhs = " << isSymbolOrConst(args[1]) << "\n"
181 );
182 break;
183 case Cmps::Mod:
184 // semi-affine-expr `mod` symbol-or-const
185 if (isSemiAffine(args[0]) && isSymbolOrConst(args[1])) {
186 LLVM_DEBUG(llvm::dbgs() << " valid affine Mod\n");
187 return TypeBinding::WithExpr(binding, BinaryOp("Mod", args));
188 }
189 LLVM_DEBUG(llvm::dbgs() << " invalid affine Mod\n");
190 break;
191 case Cmps::Neg:
192 // `-` integer-literal
193 if (isIntLit(args[0])) {
194 LLVM_DEBUG(llvm::dbgs() << " valid affine Neg\n");
195 return TypeBinding::WithExpr(binding, UnaryOp("Neg", args));
196 }
197 // XXX: May have to do tricks for negations of non literal integers like -N => 0 - N
198 LLVM_DEBUG(llvm::dbgs() << " invalid affine Neg\n");
199 break;
200 }
201
202 return TypeBinding::NoExpr(binding);
203}

221FailureOr<AffineExpr> Ctor::convertIntoAffineExpr(Builder &builder) const {
222 // If the component this expression builds is not valid return failure
223 auto tok = getToken(typeName);
224 if (failed(tok)) {
225 return failure();
226 }
227 assert(args.size() == arity(*tok));
228
229 SmallVector<AffineExpr, 2> exprs;
230 for (auto &arg : args) {
231 auto expr = arg.convertIntoAffineExpr(builder);
232 if (failed(expr)) {
233 return failure();
234 }
235 exprs.push_back(*expr);
236 }
237
238 switch (*tok) {
239 case Cmps::Add:
240 return exprs[0] + exprs[1];
241 case Cmps::Sub:
242 return exprs[0] - exprs[1];
243 case Cmps::Mul:
244 return exprs[0] * exprs[1];
245 case Cmps::Div:
246 return exprs[0].floorDiv(exprs[1]);
247 case Cmps::Mod:
248 return exprs[0] % exprs[1];
249 case Cmps::Neg:
250 return -exprs[0];
251 }
252
253 return failure();
254}
