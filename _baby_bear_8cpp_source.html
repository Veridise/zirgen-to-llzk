22Value Converter::lowerOp(
23 ExtAddOp op, ExtAddOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
24) const {
25 // Addition is an element wise addition
26 auto lhs = getTypeHelper().wrapArrayValues(adaptor.getLhs(), rewriter);
27 auto rhs = getTypeHelper().wrapArrayValues(adaptor.getRhs(), rewriter);
28 SmallVector<Value> sums;
29 sums.reserve(lhs.size());
30 std::transform(
31 lhs.begin(), lhs.end(), rhs.begin(), std::back_inserter(sums),
32 [](auto lhsVal, auto rhsVal) { return lhsVal + rhsVal; }
33 );
34 return getTypeHelper().collectValues(sums, op.getLoc(), rewriter);
35}

37Value Converter::lowerOp(

39) const {

55Value Converter::lowerOp(
56 ExtMulOp op, ExtMulOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
57) const {
58 auto a = getTypeHelper().wrapArrayValues(adaptor.getLhs(), rewriter);
59 auto b = getTypeHelper().wrapArrayValues(adaptor.getRhs(), rewriter);
60 auto NBETA = -ValueWrap(field.beta, rewriter, getTypeHelper());
61
62 auto out_0 = a[0] * b[0] + NBETA * (a[1] * b[3] + a[2] * b[2] + a[3] * b[1]);
63 auto out_1 = a[0] * b[1] + a[1] * b[0] + NBETA * (a[2] * b[3] + a[3] * b[2]);
64 auto out_2 = a[0] * b[2] + a[1] * b[1] + a[2] * b[0] + NBETA * (a[3] * b[3]);
65 auto out_3 = a[0] * b[3] + a[1] * b[2] + a[2] * b[1] + a[3] * b[0];
66
67 return getTypeHelper().collectValues(
68 ValueRange({out_0, out_1, out_2, out_3}), op.getLoc(), rewriter
69 );
70}

72Value Converter::lowerOp(
73 ExtInvOp op, ExtInvOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
74) const {
75 auto a = getTypeHelper().wrapArrayValues(adaptor.getIn(), rewriter);
76 auto BETA = ValueWrap(field.beta, rewriter, getTypeHelper());
77 auto b0 = a[0] * a[0] + BETA * (a[1] * (a[3] + a[3]) - a[2] * a[2]);
78 auto b2 = a[0] * (a[2] + a[2]) - a[1] * a[1] + BETA * (a[3] * a[3]);
79 auto c = b0 * b0 + BETA * b2 * b2;
80 auto ic = c.inv();
81 b0 = b0 * ic;
82 b2 = b2 * ic;
83
84 ValueRange arr(
85 {a[0] * b0 + BETA * a[2] * b2, -a[1] * b0 - BETA * a[3] * b2, -a[0] * b2 + a[2] * b0,
86 a[1] * b2 - a[3] * b0}
87 );
88 return getTypeHelper().collectValues(arr, op.getLoc(), rewriter);
89}

91Value Converter::lowerOp(
92 MakeExtOp op, MakeExtOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
93) const {
94 // Creating a ExtVal from a Val is creating an array and putting the Val's content into the
95 // first element. Padding with zeroes
96 auto zero = getTypeHelper().createLitOp(0, rewriter);
97 return getTypeHelper().collectValues(
98 ValueRange({adaptor.getIn(), zero, zero, zero}), op.getLoc(), rewriter
