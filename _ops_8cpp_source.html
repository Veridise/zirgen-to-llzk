45void SelfOp::build(
46 mlir::OpBuilder &builder, mlir::OperationState &state, mlir::Type compType,
47 mlir::function_ref<void(mlir::OpBuilder &, mlir::Value)> buildFn
48) {
49 state.addTypes(compType);
50 auto *region = state.addRegion();
51 region->emplaceBlock();
52 assert(region->hasOneBlock());
53 region->addArgument(compType, state.location);
54 if (buildFn) {
55 mlir::OpBuilder::InsertionGuard guard(builder);
56 builder.setInsertionPointToStart(&region->front());
57 buildFn(builder, region->getArgument(0));
58 }
59}

175template <typename CompOp> mlir::Type getSuperType(CompOp &op) {
176 if (op.isRoot()) {
177 return nullptr;
178 }
179 auto result = op.lookupFieldType(op.getSuperFieldName());
180 assert(mlir::succeeded(result));
181 return *result;
182}

184template <typename CompOp> mlir::Type getType(CompOp &op) {
185 if (op.isRoot()) {
186 return ComponentType::Component(op.getContext());
187 }
188 if (op.getParams().has_value()) {
189 auto params = *op.getParams();
190 return ComponentType::get(
191 op.getContext(), op.getSymName(), getSuperType(op), params.getValue(), op.getBuiltin()
192 );
193 } else {
194 return ComponentType::get(op.getContext(), op.getSymName(), getSuperType(op), op.getBuiltin());
195 }
196}

252 mlir::OpBuilder &builder, mlir::OperationState &state, mlir::FlatSymbolRefAttr sym,
253 uint64_t liftedParams, mlir::FunctionType fnType, bool isBuiltin
254) {
255 Properties &properties = state.getOrAddProperties<Properties>();
256
257 assert(liftedParams <= std::numeric_limits<int64_t>::max());
258 properties.numLiftedParams = builder.getIndexAttr(static_cast<int64_t>(liftedParams));
259 properties.component = sym;
260 if (isBuiltin) {
261 properties.builtin = mlir::UnitAttr::get(builder.getContext());
262 }
263
264 state.addTypes({fnType});

274mlir::LogicalResult checkConstructorTypeIsValid(
275 mlir::FunctionType expected, mlir::Type provided,
276 std::function<mlir::InFlightDiagnostic()> emitError
277) {
278 if (!mlir::isa<mlir::FunctionType>(provided)) {
279 return emitError() << "has malformed IR: was expecting a function type";
280 }
281 auto providedFn = mlir::cast<mlir::FunctionType>(provided);
282
283 if (expected.getInputs().size() != providedFn.getInputs().size()) {
284 return emitError() << "was expecting " << expected.getInputs().size() << " arguments and got "
285 << providedFn.getInputs().size();
286 }
287 if (providedFn.getResults().size() != 1) {
288 return emitError() << "was expecting 1 result and got " << providedFn.getResults().size();
289 }
290
291 // TODO: Need to check each individual type
292 return mlir::success();
293}

295mlir::LogicalResult ConstructorRefOp::verify() {
296 mlir::StringRef compName = getComponent();
297 mlir::Type type = getType();
298 auto mod = (*this)->getParentOfType<mlir::ModuleOp>();
299
300 // Try to find the referenced component.
301 auto comp = mod.lookupSymbol<ComponentInterface>(compName);
302 if (!comp) {
303 // The constructor reference could be temporarly pointing
304 // to an LLZK struct. Assume it is correct if that's the case.
305 auto structComp = mod.lookupSymbol<llzk::component::StructDefOp>(compName);
306 if (structComp) {
307 return mlir::success();
308 }
309
310 return emitOpError() << "reference to undefined component '" << compName << "'";
311 }
312
313 return checkConstructorTypeIsValid(comp.getBodyFunc().getFunctionType(), type, [&]() {
314 return emitOpError();
315 });
316}

375LogicalResult NopOp::fold(NopOp::FoldAdaptor adaptor, SmallVectorImpl<OpFoldResult> &results) {
376 if (adaptor.getIns().size() != getNumResults()) {
377 return failure();
378 }
379
380 bool atLeastOne = false;
381 for (auto in : adaptor.getIns()) {
382 if (in) {
383 atLeastOne = true;
384 results.push_back(in);
385 } else {
386 results.push_back(nullptr);
387 }
388 }
389 return success(atLeastOne);
390}

392LogicalResult NopOp::canonicalize(NopOp op, PatternRewriter &rewriter) {
393 if (op.getIns().size() == op.getNumResults()) {
394 rewriter.replaceAllUsesWith(op.getResults(), op.getIns());
395 return success();
396 }
397 return failure();

402LogicalResult SuperCoerceOp::verify() {
403 ComponentType inputType = getComponent().getType();
404 Type outputType = getVal().getType();
405
406 if (isTypeVar(outputType)) {
407 return success();
408 }
409 if (!inputType.subtypeOf(outputType)) {
410 return emitError() << "type " << outputType << " is not a valid super type of type "
411 << inputType;
412 }
413 return success();
