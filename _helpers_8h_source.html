81class GlobalBuilder {
82 mlir::ModuleOp &globalsModuleRef;
83
84public:
85 GlobalBuilder(mlir::ModuleOp &globalsModule) : globalsModuleRef(globalsModule) {}
86
87 mlir::SymbolRefAttr buildName(mlir::StringAttr flatName) const {
88 return mlir::SymbolRefAttr::get(
89 globalsModuleRef.getSymNameAttr(), {mlir::FlatSymbolRefAttr::get(flatName)}
90 );
91 }
92
93 void buildDef(mlir::Location loc, mlir::StringAttr flatName, mlir::Type type) const {
94 // This must be called for both Zhl::ConstructGlobalOp and Zhl::GetGlobalOp (because there's no
95 // requirement that a global is defined within the current compilation) so check if it already
96 // exists before creating the GlobalDefOp.
97 if (globalsModuleRef.lookupSymbol(flatName) == nullptr) {
98 mlir::OpBuilder bldr(globalsModuleRef.getRegion());
99 bldr.create<GlobalDefOp>(loc, flatName, mlir::TypeAttr::get(type));
100 }
101 }
102
103 using AndName = std::optional<std::pair<const GlobalBuilder *, mlir::StringAttr>>;
104
105 inline AndName forName(mlir::StringAttr nameOfGlobal) const {
106 return std::make_pair(this, nameOfGlobal);
107 }
108
110 inline static mlir::SymbolRefAttr
111 buildDef(const AndName &builder, mlir::Location loc, mlir::Type type) {
112 mlir::StringAttr name = builder->second;
113 builder->first->buildDef(loc, name, type);
114 return builder->first->buildName(name);
115 }
116};
