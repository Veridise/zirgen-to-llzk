181void Driver::configureLoweringPipeline() {
182 PMSetupGuard guard(pm);
183 if (emitAction == Action::PrintZHL) {
184 return;
185 }
186
187 if (emitAction >= Action::PrintLlzk || emitAction == Action::None) {
188 pm.addPass(zklang::createInjectLlzkModAttrsPass());
189 }
190 pm.addPass(zklang::createStripTestsPass());
191 pm.addPass(zml::createInjectBuiltInsPass());
192 pm.addPass(zklang::createConvertZhlToZmlPass());
193 if (!DisableCastReconciliationFlag) {
194 pm.addPass(mlir::createReconcileUnrealizedCastsPass());
195 }
196
197 if (emitAction == Action::PrintZML) {
198 return;
199 }
200
201 auto &compPm = pm.nest<zml::ComponentOp>();
202 compPm.nest<mlir::func::FuncOp>().addPass(zml::createLowerBuiltInsPass());
203 pm.addPass(zml::createRemoveBuiltInsPass());
204 pm.addPass(zml::createSplitComponentBodyPass());
205 auto &splitCompPipeline = pm.nest<zml::SplitComponentOp>();
206 auto &splitCompFuncsPipeline = splitCompPipeline.nest<mlir::func::FuncOp>();
207 splitCompFuncsPipeline.addPass(zml::createRemoveIllegalComputeOpsPass());
208 splitCompFuncsPipeline.addPass(zml::createRemoveIllegalConstrainOpsPass());
209 if (!DisableCleanupPassesFlag) {
210 splitCompFuncsPipeline.addPass(mlir::createCSEPass());
211 }
212
213 if (emitAction == Action::OptimizeZML) {
214 return;
215 }
216 pm.addPass(zklang::createConvertZmlToLlzkPass());
217 auto &llzkStructPipeline = pm.nest<llzk::component::StructDefOp>();
218 if (!DisableCastReconciliationFlag) {
219 llzkStructPipeline.addPass(mlir::createReconcileUnrealizedCastsPass());
220 }
221 if (!DisableCleanupPassesFlag) {
222 llzkStructPipeline.addPass(mlir::createCanonicalizerPass());
223 }
224}

226zirgen::dsl::ast::Module::Ptr Driver::parse() {
227 zirgen::dsl::Parser parser(sourceManager);
228 parser.addPreamble(zml::zirPreamble);
229 auto ast = parser.parseModule();
230 if (!ast) {
231 const auto &errors = parser.getErrors();
232 for (const auto &error : errors) {
233 sourceManager.PrintMessage(llvm::errs(), error);
234 }
235 llvm::errs() << "parsing failed with " << errors.size() << " errors\n";
236 return nullptr;
237 }
238 return ast;
239}

254// Simple RAII for handling the destination stream
255class Dst {
256public:
257 Dst() : dst(&llvm::outs()) {
258 std::string out = outputFile;
259 if (out.empty()) {
260 std::string base = inputFilename;
261 out = base + (emitBytecode ? ".bc" : ".mlir");
262 }
263 if (out != "-") {
264 llvm::dbgs() << "Writing result to " << out << "\n";
265 dst = new llvm::raw_fd_ostream(out, EC);
266 }
267 }
268 ~Dst() {
269 if (dst != &llvm::outs()) {
270 delete dst;
271 }
272 }
273 const std::error_code &error() { return EC; }
274
275 llvm::raw_ostream &operator*() { return *dst; }

277 void writeModule(mlir::ModuleOp mod) {
278 if (emitBytecode) {
279 mlir::BytecodeWriterConfig config;
280 // This will include the debug info as well unless `--strip-debug-info`
281 // is specified.
282 if (mlir::writeBytecodeToFile(mod, *dst, config).failed()) {
283 mod->emitOpError("could not write module bytecode to file").report();
284 }
285 } else {
286 mod->print(*dst, OpPrintingFlags(std::nullopt).enableDebugInfo(!stripDebugInfo));
287 }
288 }
289
290private:
291 std::error_code EC;
292 llvm::raw_ostream *dst;
293};

295LogicalResult Driver::run() {
296 Dst dst;
297 if (dst.error()) {
298 llvm::errs() << "Failed to open output file: " << dst.error().message() << "\n";
299 return failure();
300 }
301 openMainFile(inputFilename);
302 sourceManager.setIncludeDirs(includeDirs);
303
304 auto ast = parse();
305 if (!ast) {
306 return failure();
307 }
308 if (emitAction == Action::PrintAST) {
309 ast->print(*dst);
310 return success();
311 }
312
313 auto mod = lowerToZhl(*ast);
314 if (!mod) {
315 return failure();
316 }
317 ModuleEraseGuard guard(*mod);
318 configureLoweringPipeline();
319 LLVM_DEBUG(pm.dump());
320 if (failed(pm.run(*mod))) {
321 DEBUG_WITH_TYPE("zir-driver-dump-on-error", llvm::errs() << "Module contents:\n";
322 mod->print(llvm::errs(), OpPrintingFlags(std::nullopt).printGenericOpForm()));
323 llvm::WithColor(llvm::errs(), llvm::raw_ostream::RED, true)
324 << "An internal compiler error ocurred while lowering this module.\n";
325 return failure();
326 }
327
328 dst.writeModule(*mod);
329
330 llvm::WithColor(llvm::errs(), llvm::raw_ostream::GREEN) << "Success!\n";
331 return success();
332}

334void Driver::openMainFile(std::string filename) {
335 auto fileOrErr = llvm::MemoryBuffer::getFileOrSTDIN(filename);
336 if (std::error_code error = fileOrErr.getError()) {
337 sourceManager.PrintMessage(
338 llvm::SMLoc(), llvm::SourceMgr::DiagKind::DK_Error, "could not open input file " + filename
339 );
340 }
341 sourceManager.AddNewSourceBuffer(std::move(*fileOrErr), mlir::SMLoc());
342}
