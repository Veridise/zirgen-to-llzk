38template <const char *Name, typename CompType>
39class RemoveBuiltIn : public OpConversionPattern<CompType> {
40public:
41 using OpConversionPattern<CompType>::OpConversionPattern;
42
43 LogicalResult matchAndRewrite(
44 CompType op, typename OpConversionPattern<CompType>::OpAdaptor,
45 ConversionPatternRewriter &rewriter
46 ) const override {
47 auto name = op.getName();
48
49 // If it's not the one we are looking for just reject the match
50 if (name != Name) {
51 return mlir::failure();
52 }
53
54 rewriter.eraseOp(op);
55 return mlir::success();
56 }
57};

80class RemoveBuiltInsPass : public RemoveBuiltInsBase<RemoveBuiltInsPass> {
81
82 void runOnOperation() override {
83 auto op = getOperation();
84
85 ZMLTypeConverter typeConverter;
86 mlir::MLIRContext *ctx = op->getContext();
87 mlir::RewritePatternSet patterns(ctx);
88
89 patterns.add<
90 BitAndPattern, AddPattern, SubPattern, MulPattern, InvPattern, IszPattern, NegPattern,
91 ModPattern, ValPattern, StringPattern, ArrayPattern, InRangePattern, ExtAddPattern,
92 ExtSubPattern, ExtMulPattern, ExtInvPattern, MakeExtPattern, ExtValPattern>(
93 typeConverter, ctx
94 );
95
96 mlir::ConversionTarget target(*ctx);
97 target.addLegalDialect<
98 ZMLDialect, mlir::func::FuncDialect, index::IndexDialect, scf::SCFDialect,
99 arith::ArithDialect>();
100 target.addLegalOp<mlir::UnrealizedConversionCastOp, mlir::ModuleOp>();
101
102 target.addDynamicallyLegalOp<ComponentOp>([&](ComponentOp comp) {
103 auto found = BuiltInComponentNames.find(comp.getName().str()) != BuiltInComponentNames.end();
104 auto markedBuiltIn = comp.getBuiltin();
105 return !(found && markedBuiltIn);
106 });
107
108 if (mlir::failed(mlir::applyFullConversion(op, target, std::move(patterns)))) {
109 signalPassFailure();
110 }
111 }
112};
