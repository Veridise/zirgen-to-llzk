25class Scope {
26public:
27 enum ScopeKind {
28 Sco_Component,
29 Sco_Child,
30 Sco_Frame,
31 Sco_Block,
32 Sco_ChildEnd,
33 };
34 explicit Scope(ScopeKind Kind) : kind(Kind) {}
35 virtual ~Scope() = default;
36
37 mlir::LogicalResult declareGlobal(mlir::StringRef, TypeBinding, EmitErrorFn);
38 virtual void declareGenericParam(mlir::StringRef, uint64_t, TypeBinding) = 0;
41 virtual TypeBinding declareLiftedAffineToGenericParam(const TypeBinding &) = 0;
42 virtual void declareConstructorParam(mlir::StringRef, uint64_t, TypeBinding) = 0;
43 virtual void declareMember(mlir::StringRef) = 0;
44 virtual void declareMember(mlir::StringRef, TypeBinding) = 0;
45 virtual size_t memberCount() const = 0;
46 virtual bool memberDeclaredWithType(mlir::StringRef) = 0;
47 virtual void declareSuperType(TypeBinding) = 0;
48 virtual zirgen::Zhl::ComponentOp getOp() const = 0;
49 virtual mlir::FailureOr<TypeBinding> getSuperType() const = 0;
50 virtual TypeBinding createBinding(mlir::StringRef name, mlir::Location loc) const = 0;
51 virtual Frame &getCurrentFrame() = 0;
52 virtual const Frame &getCurrentFrame() const = 0;
53 virtual void setIsExtern() = 0;
54 virtual void setNeedsBackVariablesSupport() = 0;
55
56 ScopeKind getKind() const { return kind; }
57
58private:
59 ScopeKind kind;
60
61 static MembersMap globals;
62};
