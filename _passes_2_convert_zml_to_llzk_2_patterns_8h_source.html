137template <typename FromOp, typename ToOp>
138class LowerArithBuiltIns : public mlir::OpConversionPattern<FromOp> {
139public:
140 using mlir::OpConversionPattern<FromOp>::OpConversionPattern;
141
142 mlir::LogicalResult matchAndRewrite(
143 FromOp op, typename mlir::OpConversionPattern<FromOp>::OpAdaptor adaptor,
144 mlir::ConversionPatternRewriter &rewriter
145 ) const override {
146 rewriter.replaceOpWithNewOp<ToOp>(op, adaptor.getOperands());
147 return mlir::success();
148 }
149};
