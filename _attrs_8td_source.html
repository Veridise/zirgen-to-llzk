24def ConstExpr : ZMLAttr<"ConstExpr", "cexpr"> {
25 let summary = "constant expression";
26 let description = [{
27 Represents a constant expression that is affine representable. It is meant for translating affine maps
28 in component params from the caller's scope. Includes a list of formals that maps what other parameters from the
29 component are meant to be used as symbol arguments in the affine map.
30
31 For example `component Bar<N: Val, M: Val>(foo: Foo<N + M>)` will generate the following attribute:
32
33 ```
34 cexpr< ()[s0, s1] -> (s0 + s1), formals = 0, 1 >
35 ```
36 }];
37 let parameters = (ins "mlir::AffineMap":$map,
38 ArrayRefParameter<"uint64_t", "">:$formals);
39
40 let builders = [AttrBuilderWithInferredContext<
41 (ins "mlir::AffineMap":$map, "mlir::ArrayRef<uint64_t>":$formals), [{
42 return $_get(map.getContext(), map, formals);
43 }]>];
44
45 let assemblyFormat = "`<` $map `,` `formals` `=` $formals `>`";
46}

48def LiftedExpr : ZMLAttr<"LiftedExpr", "lexpr"> {
49 let summary = "constant expression associated to a name";
50 let description = [{
51 Represents a constant expression that is affine representable that has been associated with a parameter. It is meant for translating affine maps
52 in component params from the caller's scope. Includes a list of formals that maps what other parameters from the
53 component are meant to be used as symbol arguments in the affine map.
54
55 For example in `component Bar<N: Val, M: Val>(foo: Foo<N + M>)` the sum will get lifted into a parameter in component Bar, thus generating the following attribute for the lifted attribute:
56
57 ```
58 lexpr< @"Aff$0" : cexpr< ()[s0, s1] -> (s0 + s1), formals = 0, 1 >>
59 ```
60 }];
61
62 let parameters = (ins "::mlir::FlatSymbolRefAttr":$symbol, ConstExpr:$expr);
63
64 let builders = [AttrBuilderWithInferredContext<
65 (ins "mlir::FlatSymbolRefAttr":$symbol, "zml::ConstExprAttr":$cexpr), [{
66 return $_get(symbol.getContext(), symbol, cexpr);
67 }]>];
68 let assemblyFormat = "`<` $symbol `:` $expr `>`";
69}
