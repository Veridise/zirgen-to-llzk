61TypeBinding TypeBindings::Array(TypeBinding type, TypeBinding size, Location loc) const {
62 ParamsMap arrayGenericParams;
63 auto cleanedType = TypeBinding::StripConst(type);
64 arrayGenericParams.declare("T", cleanedType);
65 arrayGenericParams.declare("N", size);
66 TypeBinding::Flags flags;
67 flags.setSpecialized().setBuiltin();
68 TypeBinding array("Array", loc, Component(), arrayGenericParams, flags, Frame());
69 array.selfConstructs();
70 return array;
71}

100TypeBindings::TypeBindings(Location defaultLoc)
101 : unk(defaultLoc), bottom(TypeBinding(BOTTOM, unk, Component())) {
102 auto trivial = [this](StringRef name) -> const TypeBinding & {
103 return insert(name, selfConstructs(makeBuiltin(name, Component())));
104 };
105
106 auto &Val = trivial("Val");
107 auto &Type = CreateBuiltin("Type", Component());
108
109 trivial("ExtVal");
110 trivial("String");
111
112 auto T = zhl::TypeBinding::MakeGenericParam(Type, "T");
113 auto N = zhl::TypeBinding::MakeGenericParam(Val, "N");
114
115 insert(
116 "Array", selfConstructs(makeBuiltin(
117 "Array", Component(), zhl::ParamsMap().declare("T", T).declare("N", N)
118 ))
119 );
120}
