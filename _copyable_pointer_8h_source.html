30template <typename Typ, typename Factory = NullptrFactory<Typ>> class CopyablePointer {
31public:
32 using self = CopyablePointer<Typ, Factory>;
33 using type = Typ;
34 using pointer = Typ *;
35 using ref = Typ &;
36 using cref = const Typ &;
37
39 CopyablePointer() : ptr(Factory::init()) {}
40
42 CopyablePointer(pointer Ptr) : ptr(Ptr) {}
43 CopyablePointer &operator=(pointer other) {
44 safeAssign(other);
45 return *this;
46 }
47
48 CopyablePointer(const self &other) : ptr(safeCopy(other.ptr)) {}
49 CopyablePointer &operator=(const self &other) {
50 safeCopyAndDelete(other.ptr);
51 return *this;
52 }
53
54 CopyablePointer(self &&other) : ptr(other.ptr) { other.ptr = Factory::init(); }
55 CopyablePointer &operator=(self &&other) {
56 if (this != &other) {
57 safeCopyAndMove(other.ptr);
58 }
59 return *this;
60 }
61
62 ~CopyablePointer() { delete ptr; }
63
64 pointer get() { return ptr; }

80 }
81

83
85 bool operator==(const self &other) const { return ptr == other.ptr; }
86
87 self clone() const { return self(*this); }
88
89private:
90 static pointer safeCopy(const pointer other) {
91 return other ? new type(*other) : Factory::init();
92 }
93
94 void safeCopyAndDelete(const pointer other) {
95 if (ptr == other) {
96 return;
97 }
98 delete ptr;
99 ptr = safeCopy(other);
100 }
101
102 void safeCopyAndMove(pointer &other) {
103 if (safeAssign(other)) {
104 other = Factory::init();
105 }
106 }
107
108 bool safeAssign(pointer other) {
109 if (ptr == other) {
110 return false;
111 }
112 delete ptr;
113 ptr = other;
114 return true;
115 }
116
117 pointer ptr;
118};
